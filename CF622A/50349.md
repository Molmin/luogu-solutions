### 千万不要用数组会 MLE 的


下面直接贴代码。


```
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll n,x=1,y;
// x 上一个数, y 当前的数。 
int main()
{
	cin>>n;
	if(n==1) {cout<<1<<endl; return 0;}
	//特殊判断 
	for(ll i=2;i<=50000000;i++)
	{
   	//for循环内一定要开ll，不然会爆的。
		y=i*(i-1)/2+1;
    //第i个1 的位置。
		if(y>=n)
		{
			if(y==n) {cout<<1<<endl; return 0;}
            // 1的位置，直接输出。
			else {cout<<n-x+1<<endl; return 0;}
            // 注意：一定要+1.
		}
		x=y;
	}
	return 0;
}
```
