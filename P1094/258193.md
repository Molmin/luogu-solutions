这题因为要使分组最少且每组最多两个，所以要尽可能的使每组的总价值接近上限。

读入之后，用sort从小到大排列，两个指针分别指向首尾，分组之后往中间走。

代码如下：
```cpp
#include<bits/stdc++.h>//万能头
using namespace std;

int main(void)
{
	int w,n,i,j,t=0,ans=0;//t表示已经用掉的纪念品数，ans表示分组数
	cin>>w>>n;
	int an[30005];//稍微开大点
	for(i=1;i<=n;i++){//读入
		cin>>an[i];
	}
	sort(an+1,an+n+1);//排列
	i=1;//分别指向首尾
	j=n;
	while(t<n&&i<j){//分完的小于总数且不指向同一个
		if(an[i]+an[j]<=w){//如果两者和小于上限，则这两个分一组，指针往中间走
			i++;
			j--;
			ans++;
			t+=2;
		}
		else{//如果超过上限，后面的一个单独分一组，前面的等待下一个
			j--;
			ans++;
			t++;
		}
	}
	if(t<n){//如果t<n，说明中间剩着一个，单独分一组
		ans++;
	}
	cout<<ans;
	return 0;
}
```
