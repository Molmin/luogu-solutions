**本题思路：**

先枚举出在 $720$ 分钟内有几个等差数列的时刻，分别要过多少分钟，并用 $a$ 数组把它们存起来。

对每次输入的数取模于 $720$，在乘 $31$,最后算出剩下的大于 $a$ 数组中的数据个数，相加就可以了。

**下面是代码。**

```c
#include<iostream>
using namespace std;
#define ll long long 
ll n=0,ans=0;
int main()
{
    cin>>n;
    int a[31]={34,
    71,83,95,107,119,
    130,142,154,166,178,
    201,213,225,237,
    260,272,284,296,
    331,343,355,
    390,402,414,
    461,473,
    520,532,
    591,
    671};
    for(ll i=0;i<n;i++)
    {
        ans=0;
        int a1;
        scanf("%d",&a1);
        if(a1>720)
        {
            ans+=a1/720*31;
            a1=a1%720;
        }
        for(int j=0;j<31;j++)
        {
            if(a1>=a[j])
            {
                ans++;
            }
            else
            break;
        }
        printf("%lld\n",ans);
    }
    
    return 0;
}
```
