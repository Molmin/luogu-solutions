主要就是把文章的单词分开然后给每个单词一个特征值，边输单词边匹配，就是这么简单的思路的说.......(dalao求轻虐....  
```cpp
#include <iostream>
#include <math.h>
#include <stdio.h>
using namespace std;
int main()
{
   char s='0';//临时的变量....
   long long a=0,time=0,Y=1000100,n=0,G=0,target=1,temp=1;//time是出现的次数，Y是首次出现的地方
   for(;;){                     //target是目标的“值”，temp是每个单词的“值”，其他的是计数器~~
       s=getchar();              //第一个for是依次读取输入的字母（同时转小写）
       if(s=='\n') break;
       s=tolower(s);
       target+= s*(pow(10,n));   //然后用n做计数器，相当于把单词变成asc2码组成的值~~~如AB就是6665
       n+=2;                     //（不过当asc2码大于一百就不是单纯的组合形式了...嘛只要每个单词能确定唯一一个值就好惹）
   }
   n=0;                          //n你完成任务了，继续为下个循环工作吧！
   while(s!=EOF ){
       s=getchar();                  //第二个循环就是文章输入的说
       if(s!=' '&&s!='\n'&&s!=EOF){ //通过输入空格换行等来判定单词的结束
        s=tolower(s);
        temp+=s*(pow(10,n));         //边输字母边依次给输入的单词编值~
        n+=2;                        //
       }                             //
       else{
        if(temp==target){         //输入完空格就来判断与目标值是否相等
            if(G<Y) Y=G;        //标记首个出现的单词
              time++;            //相等就计数
            }
        G= a+1;                   //G表示下个单词的首字母位置（一开始题看错了以为每个相同单词的位置都要输出..懒得改了~~）
        temp=1;                   //temp继续初始化，以计算下个单词的值
        n=0;                      //n也要初始化
        }
        a++;                    //a表示有多少字符（确定位置
   }
   if(time!=0) cout<<time<<" "<<Y; //然后就是输出了~~~~
   else cout<<"-1";
}
```
