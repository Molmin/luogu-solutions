我们要满足题意，只需要把 $XOR$ 与 $XOR + SUM$ 丢入即可，即：    
```
XOR ^ XOR ^ (XOR + SUM) = (XOR + XOR + SUM + SUM) / 2
```
此题完，我承认我r $Trie$ r 了一个世纪。    
```cpp
#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
const int Len = 1e5 + 5;
long long n,m,XOR,SUM,x;
int main()
{
	int T;
	scanf("%d",&T);
	while(T --)
	{
		XOR = SUM = 0;
		scanf("%lld",&n);
		for(int i = 1 ; i <= n ; i ++)
		{
			scanf("%lld",&x);
			XOR ^= x;
			SUM += x;
		}
		printf("2\n");
		printf("%lld %lld\n",XOR , XOR + SUM);
	}
	return 0;
}
```
