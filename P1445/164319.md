$$
\frac{x+y}{xy}=\frac{1}{n!}
$$

$$
xy-n!(x+y)=0
$$

$$
(n!)^2+xy-n!(x+y)=(n!)^2
$$

$$
(x-n!)(y-n!)=(n!)^2
$$

$$
a=(x-n!),b=(y-n!)
$$

$$
ab=(n!)^2
$$

 n!=p_1^{c_1}*p_2^{c_2}*...p_k^{c_k}
