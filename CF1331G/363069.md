输出样例骗点分吧。竟然过了！

原来，这是CF的愚人节题目啊！题目无需多说，上代码：
```cpp
#include<bits/stdc++.h>
using namespace std; 
int main() 
{ 
	printf("f(10) = MAGNA NIMIS!\n");//不需输入，直接输出
	printf("f(-9) = -3642.00\n");//记得换行
	printf("f(-8) = -2557.17\n");
	printf("f(-7) = -1712.35\n");
	printf("f(-6) = -1077.55\n");
	printf("f(-5) = -622.76\n");
	printf("f(-4) = -318.00\n");
	printf("f(-3) = -133.27\n");
	printf("f(-2) = -38.59\n");
	printf("f(1) = 6.00\n");
	printf("f(0) = 0.00\n");
	return 0;//完结撒花
} 
```
