解决方案：输样例

话不多说，上AC代码

```cpp
#include <iostream>
using namespace std;
int main()
{
   cout <<"f(10) = MAGNA NIMIS!\n";//注意要带换行符\n
   cout <<"f(-9) = -3642.00\n";
   cout <<"f(-8) = -2557.17\n";
   cout <<"f(-7) = -1712.35\n";
   cout <<"f(-6) = -1077.55\n";
   cout <<"f(-5) = -622.76\n";
   cout <<"f(-4) = -318.00\n";
   cout <<"f(-3) = -133.27\n";
   cout <<"f(-2) = -38.59\n";
   cout <<"f(1) = 6.00\n";
   cout <<"f(0) = 0.00"; 
   return 0;//完美结局
}
```
