```cpp
# include <bits/stdc++.h>
using namespace std;
int main() {
    string s; //定义一个字符串
    getline(cin, s); //使用getline读入字符串
    int len = s.size(); //使用s.size()函数得出字符串的长度
    int ans = 0;
    for(int i = 0; i < len; i ++) //枚举字符串中的每一个元素
        if(s[i] != ' ') ans ++; //因为回车不会读入进去（这是getline的特性），所以只需要判断是不是空格即可
    printf("%d", ans); //输出答案
}
```
