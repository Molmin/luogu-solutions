首先，要注意题面给了一句话“注:除最后一个波形外，每个单独的波形后面都有一个空行。”就是综上所述，每一个波形之间都一个换行两次，也就是空出一行，而只有最后一个波形后面一定不能多一行。

注意到这个事情就真的是入门难度了。


```cpp

#include<bits/stdc++.h>//万能头
using namespace std;
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		int a,f;
		cin>>a>>f;//a代表振幅，是波的高度；f代表频率，也就是波的个数
		while(f--){//输出f个波，当f==0时循环结束
		    for(int j=1;j<=a;j++)//第一个循环输出前一半波
		    {
			    for(int k=1;k<=j;k++)
			    cout<<j;
			    cout<<endl;
		    }
		    for(int j=a-1;j>=1;j--)//第二个循环输出后一半波，因为上一个循环已经到a了，所以跳过a，从a-1往回输
		    {
			    for(int k=1;k<=j;k++)
			    cout<<j;
                cout<<endl;
		    }
		    if(i!=n||f!=0)//i!=n:指不是最后一个样例 f!=0:指不是最后一个波形 任意满足就回车，如果都不满足就不回车
            cout<<endl;//唯一的坑点：最后一个只有一个回车
	    }
	}
}
```
