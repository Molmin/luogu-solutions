着实是一道水题，用条件即可，甚至无需字符串！！！太简单了，上代码
```
#include<bits/stdc++.h>
using namespace std;
int main()
{char a[12];
cin>>a[1]>>a[2]>>a[3]>>a[4]>>a[5]>>a[6]>>a[7]>>a[8];\\输入
if(a[3]=='B')\\核心判断
cout<<"Bachelor"<<" ";
else if(a[3]=='M')
cout<<"Master"<<" ";
else
cout<<"Doctor"<<" ";
cout<<a[1]<<a[2]<<endl;\\直接输出第一和第二字符
return 0;
}
```
