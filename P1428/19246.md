使用c++编程


```cpp
#include<iostream>
using namespace std;
int main()
{
    int i,j,k=0,n;  //n为我们输入的一个数，k=0用来计算第n个数比前面数大的个数，不过全局变量会是结果变化，如果你也是初学者，可以试试
    cin>>n;
    int a[n];     //定义一个数组用来存放数据
    for(i=0;i<n;i++)
    {
        cin>>a[i];     //使用for循环依次将输入的数储存在数组中
    }
    cout<<"0 ";     //因为无论第一个数为多少，其输出结果都是0，所以我们可以直接输出
    for(i=1;i<n;i++)    
       {  int k=0;                       //局部变量k=0
             for(j=0;j<=i-1;j++)
             if(a[i]>a[j])               //进行比较
             k++;                            
             cout<<k<<" ";
       }
    return 0;
}
```
