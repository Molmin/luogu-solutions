```cpp
#include <cstdio>//头文件
int a[10001];//开一个数组来存储输入的数据
int main()
{
    int n=0,l=0,f=0,c=0;//n表示这是一个n行n列的矩阵，l表示有l个指令，f表示当前在第几列，c表示当前要输出的数字（color 颜色）
    scanf("%d",&n);//输入n表示这是一个n行n列的矩阵
    while(scanf("%d",&a[++l])!=EOF);//直到读入到文件结束为止
    for(int i=1;i<=l;i++)//执行这l个指令（即l代表指令的个数）
    {
        for(int j=1;j<=a[i];j++)//这个指令的长度
        {
            printf("%d",c);//输出当前要输出的数字
            f++;//前进一个
            if(f%n==0)//如果到达了行尾
            {
                printf("\n");//打印回车
   //也可以写一句：f=0;
            }
        }
        if(c==0)//更改当前要输出的数字（即更改颜色）
        {
            c=1;//更改当前要输出的数字为1
        }
        else//否则
        {
            c=0;//更改当前要输出的数字为0
        }
    }
    return 0;//结束程序
}
```
