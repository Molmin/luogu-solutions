这题我的思路就用一个变量判断奇偶，然后按题目给出的压缩码输出，当奇偶变量为1时输出1，否则输出0即可，这题的唯一难点就是换行问题。

解决换行问题，就用一个变量存储已经输出汉字点阵图数的数量，当这个变量时n的倍数时则代表这一行已经输完，需要换行，而且第一次为0时不能换行（细节），因为还没开始。

附上代码：
    
    #include<iostream>   //文件头
    using namespace std;
    int n,a,g,t;      //定义，a表示0，1的压缩码数，g表示汉字点阵图的个数+判断换行，t判断0，1
    int main()
    {
        cin>>n;       //输入n
        while(g<n*n)   //当g还没到n*n，还没有输出完时执行
        {
            cin>>a;   //读入1或0的连续个数
            for(int i=1; i<=a; i++)  //有a个0或1就循环处理a次
            {
                if(g%n==0&&g!=0)cout<<endl;  //换行问题，当g是n的倍数时就代表这一行已经输完，需要换行，而且第一次g==0时不能换行（细节）
                if(t%2==0)cout<<0; else cout<<1; //根据算出的奇偶变量输出点阵图
                g++;        //点阵图个数++
            }
            t++;  //判断奇偶变量++
        }
        return 0;
    }
