理理这题的思路：
首先，输入是无限输入的。
接着，我们就要判断一下这是第奇数个输入还是第偶数个输入，因为奇数输出a个零，偶数输出a个1。然后，我们需要用一个pos变量来表示有没有到换行的时机。最后，没有最后了。

不太华丽的分割线

&&--------$$$---------***----------$$$---------&&

```cpp
#include<bits/stdc++.h>
using namespace std;
int a,n,pos=0,ssa=1;
int main()
{
	cin>>n;
	while(cin>>a){
		if(ssa%2!=0)//如果a是第奇数个
        {
			for(int k=1;k<=a;k++)
            {
				cout<<0;//输出0
				pos++;
				if(pos==n)//判断有没有输出到第n个
                {
					cout<<endl;//换行
					pos=0;//清零
				}
			}
		}
		else if(ssa%2==0)//如果a是第偶数个
        {
			for(int l=1;l<=a;l++)
            {
				cout<<1;//输出一
				pos++;
				if(pos==n)//判断有没有到换行的时机
                {
					cout<<endl;//换行
					pos=0;//清零
				}
			}
		}
		ssa++;//a的位置变化一下
	}
	return 0;//it's very important!!!
}
```
