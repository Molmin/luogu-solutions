# 这种方法简单实用

直接上代码

```cpp
#include<iostream>
using namespace std;
int n/*表示n*n的矩阵*/,x/*每次读入的0或1的个数*/,tot=0/*记录输出位置，目的是记录换行*/,sum=0/*已输出了多少字符，退出循环的必要条件*/;
int main(){
	cin>>n;
	for(int i=1;i;i++){
		cin>>x;
		sum+=x;
		for(int j=1;j<=x;j++){
			tot++;
			if(tot==n+1){    /*输入到最后了，换行！*/
				tot=1;
				cout<<endl;
			}
			if(i%2){            /*该输出0了?*/
				cout<<"0";
			}
			else cout<<"1";
		}
		if(sum>=n*n)break;    /*字符已够，退出循环*/
	}
} 
```
