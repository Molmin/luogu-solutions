```C
	for(i=0;i<M;i++)W[i]=-1;
```

```C
	for(i=0;i<N;i++){
		scanf("%d",&w),q=1;
		for(p=0;p<M;p++){//检索 
			if(w==W[p]){
				q--;break;
			}		
		}
		if(q==0)continue;//内存中存在则跳过 
		else{
			W[r]=w;n++;r++;
			if(r==M)r=0;//否则进行查找并存储，注意，这里巧妙地实现了先进先出：r来确定目前轮到谁进谁出
		}
	}
	printf("%d",n);
	return 0;
}
```
