一开始用0来判断，结果发现好几个数据中都有0（害我找了好半天/(ㄒoㄒ)/~~），于是改用-1来判断，瞬间通过

下面奉上代码：

```cpp
#include<bits/stdc++.h>//万能头文件
using namespace std;//定义标示符
int m,n,i,j,s,g;//定义变量
bool b;
int main()//定义主程序
{
        cin>>m>>n;//读入内存容量和文章的长度
        int a[n]={0},r[m]={0};//定义内存和文章
        for(i=0;i<n;i++)//读入文章
                cin>>a[i];
        r[m-1]=-1; //将内存的最后一位设为-1
        for(i=0;i<n;i++)
        {
                b=false;//初始化b
                for(j=0;j<g;j++)//从内存中找
                        if(a[i]==r[j])//如果找到了这个单词
                        {
                                b=true;//b变为真
                                break;//退出循环
                        }
                if(b)//如果刚才找到了
                        continue;//退出循环
                s++;//外查字典次数+1
                if(r[m-1]==-1)//如果内存未满（最后一位还未使用）
                {
                        r[g]=a[i];//在内存中添加此词
                        g++;
                }
                else//否则（内存已满）
                {
                        for(j=0;j<m-1;j++)//把第一个储存的替换掉
                            r[j]=r[j+1];
                        r[m-1]=a[i];//将最后一个设为此词
                }
        }
        cout<<s;//打印总外查字典次数
        return 0;
}
```
