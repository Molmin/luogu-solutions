打表CODE

```cpp
#include<cstdio>//这个头文件比iostream快，内存也小 
using namespace std;
int main()
{
	int sum=0;//计数器初始化为零 
	for(int i=1;i<=100;i++)//计算1-100的所有和 
	{
		for(int j=1;j<=i;j++)//计算单个数即i的和(所以j<=i) 
			sum+=j;//计数器累加
		//美观程序分界线 
		if(i==100)//如果达到最后一位即100 
		{
			printf("%d",sum);//就不打印逗号 
			return 0;//直接结束 
		}
		//美观程序分界线 
		printf("%d,",sum);//打印累加和与逗号(为的是复制方便) 
		sum=0;//计数器归为0，这步一定要写！！(作者第一次就是这么WA的) 
	}
	return 0;
}
```
