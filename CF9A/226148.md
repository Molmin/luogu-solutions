这个题思路实际上是，找到小Y和小W中的最大值x，然后用6-x和6约分，但我根本就不知道什么GCD函数。只能用蒟蒻的方法。

```
#include <bits/stdc++.h>
using namespace std;
long long a,b,c,d;
int main()
{    
     for(int a=1;a<=6;a++)//从1到6枚举
           for(int b=1;b<=6;b++)
 {
    c=6-max(a,b)+1;//寻找，6-这个值
    d=6;
    for(int i=2;i<=6;i++)
    while(c%i==0&&d%i==0)//如果能除尽，就使劲往下除
	{
   c/=i;
	 d/=i;	}
	 cout<<'"'<<c<<"/"<<d<<'"'<<","<<endl;
    }
	return 0;
}
```
