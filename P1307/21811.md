这道题可利用字符数组字符串来达到目的，输入字符串后，先除去后面的‘\0'，再对正负数分别进行'对称变换'，最后输出即可：

```cpp
#include<stdio.h>
#include<string.h>
int main()
{
    char a[99];                 //定义字符数组 
    int b,i,j;
    scanf("%s",a);              //输入定为字符串存在数组中 
    for(i=97;i>0;i--)
     if(a[i]=='0'&&a[i+1]=='\0')a[i]='\0';//去末尾的0 
    j=strlen(a);                //通过存放长度来解决对称需要 
    if(a[0]=='-')               //讨论负号 
     for(i=1;i<=j/2;i++)
      {b=a[i];a[i]=a[j-i];a[j-i]=b;}//进行“对称翻转” 
    else for(i=0;i<j/2;i++)
          {b=a[i];a[i]=a[j-i-1];a[j-i-1]=b;}//同上 
    puts(a);                     //输出结果 
    return 0;
}
```
