看到这题的第一眼，我的设想是用字符串，把首尾对应的项交换即可，但这个题有负数，于是就有了我下面这种~~脑回路清奇的~~正负数判定法
```
#include<iostream>
#include<cstring>
using namespace std;
char s[20];//先定义个字符数组 
int main()
{
	int t,x=0;//t用来交换数字，x用来判定是正数还是负数 
	cin>>s;//输入这个数（废话） 
    for(int i=0;i<=strlen(s)/2+x-1;i++)//x的值不同，循环终止条件也不同
    {
    	if(s[i]=='-')//判定 
    	{
			x=1;//神奇的x，它代表了正or负两种不同情况 
			continue;
		}
		t=s[i];//重点来了 （敲黑板） 
		s[i]=s[strlen(s)-i-1+x];//正负数不同，数组第i项对应的项的项数也不同 
		s[strlen(s)-i-1+x]=t;//完成数的交换 
	}//原数转换完毕，但开头有可能是0 
	while(s[x]=='0')//神奇的x又出现了 
	{
		for(int i=x;i<=strlen(s)-2;i++)//哪里是0，就从哪里开始处理 
			s[i]=s[i+1];//把开头的0用后一个数字替代 
		s[strlen(s)-1]=' ';//把最后一项换成空格 
	}
	cout<<s;//输出这个数（还是废话） 
    return 0;
}
```
