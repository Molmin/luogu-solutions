```cpp
//写的都有些像伪代码了，2333
//因为是for循环，所以特别直观，相比别的代码，这个代码则是最令初学者看懂的了
//代码如下:
#include <iostream>//输入输出流
using namespace std;//命名空间
int main(){//主函数
    int a,b,i,sum=0;//定义整形变量
    cin>>a>>b;//读入星期几，过了的天数[按照题目要求]
    for (i=a;i<=a+b-1;++i){//这则是很重要的了，注意为什么是i<=a+b-1
        //为什么i=a+b-1，因为包涵了第一天，也算一天
        if (i%7==6 || i%7==0){//别上当，我第一次主观臆断，写的就是if(i%6==0&&i%7==0)，
                              //因为是i，不是i所对应的星期几，程序时运行关于i的
            continue;//跳过用continue,千万不要用break或return 0;不然就把整个函数都给停掉了
        }
        sum+=250;//没被跳过的，就按要求加上每日的250就可以了:-)
    }
    cout<<sum<<endl;//输出答案
    return 0;//现在用通过函数返回值，正常地停掉函数，也就是结束着整个程序
```
