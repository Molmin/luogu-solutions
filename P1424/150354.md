表示这一道题可以用着一种求余数和整除的方法来解决。
详细的在代码里面：
```cpp
#include<bits/stdc++.h>  //不想打那么多头文件，一个万能头搞定。
using namespace std;
int n,m,s;
int main()
{
    cin>>n>>m; //输入两个数
    s+=m/7*1250; //直接计算这只鱼要游几周，一周这只鱼要游1250km
    m%=7; //因为把他的要游几周算出来了，所以就剩下那一单天了。
    for(int i=n;i<n+m;i++)//for循环找那些身下来的天。  
        if(i%7!=0 && i%7!=6) s+=250; //s%7怕剩下的天数累计后会超过7.        
    cout<<s; //输出。    
    return 0;
}
```
