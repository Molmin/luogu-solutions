题目要求每位顾客的菜是不同的，于是：

>我们可以把 $a,b,c$ 中大于 $4$ 的变成 $4$
>
>而且答案与 $a,b,c$ 的顺序无关（想想为什么？）

于是果断打表，依题意手玩即可~

```cpp
#include<cstdio>
#include<algorithm>

int ans[5][5][5];

// 打表
void init()
{
	ans[0][0][0]=0;
	ans[0][0][1]=1;
	ans[0][0][2]=1;
	ans[0][0][3]=1;
	ans[0][0][4]=1;
	ans[0][1][1]=2;
	ans[0][1][2]=2;
	ans[0][1][3]=2;
	ans[0][1][4]=2;
	ans[0][2][2]=3;
	ans[0][2][3]=3;
	ans[0][2][4]=3;
	ans[0][3][3]=3;
	ans[0][3][4]=3;
	ans[0][4][4]=3;
	ans[1][1][1]=3;
	ans[1][1][2]=3;
	ans[1][1][3]=3;
	ans[1][1][4]=3;
	ans[1][2][2]=4;
	ans[1][2][3]=4;
	ans[1][2][4]=4;
	ans[1][3][3]=4;
	ans[1][3][4]=4;
	ans[1][4][4]=4;
	ans[2][2][2]=4;
	ans[2][2][3]=5;
	ans[2][2][4]=5;
	ans[2][3][3]=5;
	ans[2][3][4]=5;
	ans[2][4][4]=5;
	ans[3][3][3]=6;
	ans[3][3][4]=6;
	ans[3][4][4]=6;
	ans[4][4][4]=7;
}

int main()
{
	init();
	int T;
	scanf("%d",&T);
	while(T--)
	{
		int a,b,c;
		scanf("%d%d%d",&a,&b,&c);
		if(a>4)a=4;if(b>4)b=4;if(c>4)c=4;
		// 排序
		if(a>b)std::swap(a,b);if(a>c)std::swap(a,c);if(b>c)std::swap(b,c);
		printf("%d\n",ans[a][b][c]);
	}
	return 0;
}
```
