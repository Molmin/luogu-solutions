显然暴力DP。

因为只能向上，向右，向左走，所以左边不会被改变。

那么左边就可以确定了。

那么状态转移方程就很好确定了。

$dp[i][j]=max(dp[i][j-1],dp[i-1][j],dp[i+1][j])$

因为确定的只有$dp[i][j-1]$，那么我们就用其他数组表示$dp[i-1][j]$和$dp[i+1][j]$。

上代码：

```
#include<iostream>
#include<cstring>
#include<cstdio>
#include<cmath>
using namespace std;
long long flag[2000][2000],flag2[2000][2000];
long long dis[2000][2000],dp[2000][2000];
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			cin>>dis[i][j];
	dp[1][1]=dis[1][1];
	for(int i=2;i<=n;i++)
		dp[i][1]=dis[i][1]+dp[i-1][1];//初始化 
	for(int j=2;j<=m;j++)
	{
		flag[1][j]=dp[1][j-1]+dis[1][j];
		flag2[n][j]=dp[n][j-1]+dis[n][j];
		for(int i=2;i<=n;i++)
			flag[i][j]=max(flag[i-1][j],dp[i][j-1])+dis[i][j];//从上往下跑一遍 
		for(int i=n-1;i>=1;i--)
			flag2[i][j]=max(flag2[i+1][j],dp[i][j-1])+dis[i][j];//从下往上跑一遍 
		for(int i=1;i<=n;i++)
			dp[i][j]=max(flag[i][j],flag2[i][j]);//取最大值 
	}
	cout<<dp[n][m];
	return 0;
}
```
