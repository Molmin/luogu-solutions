#include<bits/stdc++.h>
using namespace std;
int main()
{
    int a[4];
    for(int i=1;i<=3;i++)
    cin>>a[i];//读入三条边 
    sort(a+1,a+1+3);//找出最小边和最大边，因为在三角形小角对小边，斜边无疑是最大的边了 
    for(int j=1;j<=a[1];j++)//枚举除数，只要到这边的最小边就行了，如果大于最小边，就不能化简了 
    {
        if(a[1]%j==0&&a[3]%j==0)//这里找的是最小公因数，如果都能把j整除，就将最小边和斜边的值都除以这个j； 
        {
            a[1]/=j;//更新小边的值 
            a[3]/=j;//更新大边的值 
            j=1;//重新枚举 
        }
    }
    cout<<a[1]<<"/"<<a[3];//最后输出就行了 
}
