		for (ll i = 0;i<this->size;i++)
			for (ll j = 0;j<this->size;j++)
				for (ll k = 0;k<this->size;k++)
					ans.values[i][k] += this->values[i][j]*other.values[j][k];
		return ans;
	}
	
	Matrix operator ^(ll b) {
		Matrix ans(this->size);
		Matrix base = *this;
		for (ll i = 0;i<this->size;i++)
			ans.values[i][i] = 1;
		while(b) {
			if (1&b)
				ans = ans*base;
			base = base*base;
			b>>=1;
		}
		return ans;
	}

		for (ll i = 0;i<this->size;i++)
