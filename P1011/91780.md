这道题，名义上是普及-，实际上它是一道美丽（~~丑陋~~）的数论。

首先，我们需要找规律（设第二站上下车x人）：

|  | 1 | 2 | 3 | 4 | 6 | 7 | 8 |
| -----------: | -----------: | -----------: | -----------: | -----------: | -----------: | -----------: | -----------: | -----------: |
| 上车人数（u数组）| a | x | a+x | a+2x | 2a+3x | 3a+5x | 5a+8x | 
| 下车人数 | 0 | x | x | a+x | a+2x | 2a+3x | 3a+5x |
| 车上人数（d数组） | a | a | 2a | 2a+x | 3a+2x | 4a+4x | 6a+7x


可以很清楚的推出d[i]与下车人数一点关系都没有！
递推公式是：d[i]=u[i-2]+d[i-1]。

然而，在编程中，这个未知数x该怎么解决呢？

#include<iostream>
#include<cstdio>
#include<string>
using namespace std;
string u[25],d[25];  //如上图所示，注意是字符串数组。字符串数组怎么操作我就不讲了。 
int main()
{
	freopen("station.in","r",stdin);
	freopen("station.out","w",stdout); 
	int a,n,m,x,la=0,ll=0;  //定义，后面会用到，la和ll是计数器，要清零。 
	cin>>a>>n>>m>>x;
	n--;  //根据题目意思，第1站的人数不需要推，所以只要推n-1次。 
	u[1]='a',u[2]='l';  //重点来了（圈起来，考试要考的！)，用字符串数组模拟设未知数的过程，由于题目中有x出现，所以图中的x用l代替。 
	d[1]='a',d[2]='a';  //这两行是赋初值。 
	for (int i=3;i<=n;i++)
		u[i]=u[i-1]+u[i-2],d[i]=d[i-1]+u[i-2];  //递推过程，从上图推出来的。 
	for (int i=0;i<d[n].size();i++)  //这个循环，统计a与l的个数，让这题用方程思想做出来。 
		if (d[n][i]=='a')
			la++;  //la表示a的个数。 
		else
		if (d[n][i]=='l')
			ll++;  //ll表示l的个数。 
	int r=(m-la*a)/ll;  //计算l的值，记为r。 
	la=ll=0;  //下面还要用，先清零。 
	for (int i=0;i<d[x].size();i++)  //这个循环与上面的类似，只是把d[n]改为了d[x]。 
		if (d[n][i]=='a')                                             
			la++; 
		else
		if (d[n][i]=='l')
			ll++; 
	cout<<la*a+ll*r<<endl;  //计算第x站开出是车上的人数并输出。 
	fclose(stdin);
	fclose(stdout);
	return 0;
}
