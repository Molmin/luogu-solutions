#### 这道题非常简单，只需要用一个桶排的思想就能做出来了，题目意思是，把他所有看到的车站号码记录下来，如果这些车站每次都被他看见了，就输出这些车站。也就是说只需要把他所有看到的车展放到桶里，如果当你最后一次放的时候发现这个桶已经放过n次了，就把他存起来（注意不要直接输出），要排序。

下面上代码
```cpp
#include<bits/stdc++.h>
using namespace std;
int p[100010];
int a[100010];
int main()
{
	int n,o=0,l=0;
	cin>>n;
	l=n;//要用l来保存n，因为n在循环中会改变
	memset(a,0,sizeof(a));//注意清零
	while(n--)
	{
		int m,k;
		cin>>m;
		for(int i=1;i<=m;i++)
		{
		cin>>k;
		a[k]++;//桶排基本操作
	    if(a[k]==l)//每次都看到了这个数
	    {
	    	p[++o]=k;//注意是先加再去用++o，不是o++
		}
		}
	}
	sort(p+1,p+o+1);//排一下序
	for(int i=1;i<=o;i++)
	cout<<p[i]<<" ";//输出
}

```
