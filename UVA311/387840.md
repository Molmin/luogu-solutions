### 1 思路

这就是一道大模拟。

设 $i\times i$ 的物品有 $a[i]$ 个，设需要 $ans$ 个包装袋。

首先，$6\times6$ 的物品是要单独一个包装袋的，所以初始时 $ans=a[6]$，然后将 $a[6]$ 设为 $0$。

然后分别对于从 $i=5$ 到 $i=1$ 的情况**循环**。

+ 当 $a[5]$ 还大于 $0$ 时：

	$a[5]$ 自减 $1$，$ans$ 自增 $1$。
    
   将剩下的 $11$ 个格子全部填上 $1\times1$ 的物品，即：
   
   + 如果 $a[1]\ge11$，则 $a[1]$ 自减 $11$。
   
   + 如果 $a[1]<11$，则将 $a[1]$ 设为 $0$。
   
+ 当 $a[4]$ 还大于 $0$ 时：

	$a[4]$ 自减 $1$，$ans$ 自增 $1$。
    
   + 如果 $a[2]\ge5$，则 $a[2]$ 自减 $5$。
   
   + 如果 $a[2]<5$，则将 $a[2]$ 设为 $0$。并且，将剩下的格子全部填满 $1\times1$ 的物品，记得更新 $a[1]$。
   
+ 当 $a[3]$ 还大于 $0$ 时：

	$ans$ 自增 $1$。

	+ 如果 $a[3]\ge4$，则 $a[3]$ 自减 $4$。
    
   + 如果 $a[3]=3$，则将 $a[3]$ 设为 $0$。
   
   	 	+ 如果 $a[2]\ge1$，则 $a[2]$ 自减 $1$，将剩下的 $5$ 个格子全部填满 $1\times1$ 的物品。
        
     	+ 如果 $a[2]=0$，则将剩下的 $9$ 个格子全部填满 $1\times1$ 的物品。
     
   + 如果 $a[3]=2$，则将 $a[3]$ 设为 $0$。
   
   		+ 如果 $a[2]\ge3$，则将 $a[2]$ 自减 $3$，剩下的 $6$ 个格子全部填满 $1\times1$ 的物品。
        
    	 + 如果 $a[2]<3$，则将 $a[2]$ 设为 $0$，剩下的格子全部填满 $1\times1$ 的物品。
         
   + 如果 $a[3]=1$，则将 $a[3]$ 设为 $0$。
   
   		+ 如果 $a[2]\ge5$，则将 $a[2]$ 自减 $5$，剩下的 $5$ 个格子全部填满 $1\times1$ 的物品。
        
     	+ 如果 $a[2]<5$，则将 $a[2]$ 设为 $0$，剩下的格子全部填满 $1\times1$ 的物品。
      
  同样记得更新好 $a[1]$。
     
+ 当 $a[2]$ 还大于 $0$ 时：

	$ans$ 自增 $1$。
    
   + 如果 $a[2]\ge9$，则将 $a[2]$ 自减 $9$。
   
   + 如果 $a[2]<9$，则将 $a[2]$ 设为 $0$，剩下的格子全部填满 $1\times1$ 的物品。记得更新好 $a[1]$。
   
+ 当 $a[1]$ 还大于 $0$ 时，对于每 $36$ 个 $a[1]$，新增一个包装袋，让 $ans$ 自增 $1$。不足 $36$ 的部分视为 $36$。

### 2 代码与记录

```cpp
#include<cstdio>
using namespace std;
int a[8];
int ans;
int main(){
	#ifndef ONLINE_JUDGE
	freopen("UVA00311_1.in","r",stdin);
	freopen("UVA00311_1.out","w",stdout);
	#endif
	while(true){
		for(int i=1;i<=6;++i){
			scanf("%d",a+i);
		}
		if(a[1]==0&&a[2]==0&&a[3]==0&&a[4]==0&&a[5]==0&&a[6]==0)break;
		ans=a[6];
		a[6]=0;
		while(a[5]>0){
			++ans;
			--a[5];
			if(a[1]>=11)a[1]-=11;
			else a[1]=0;
		}
		while(a[4]>0){
			++ans;
			--a[4];
			if(a[2]>=5)a[2]-=5;
			else{
				int i=20-4*a[2];
				if(a[1]>=i)a[1]-=i;
				else a[1]=0;
				a[2]=0;
			}
		}
		while(a[3]>0){
			++ans;
			if(a[3]>=4)a[3]-=4;
			else{
				if(a[3]==3){
					if(a[2]>=1){
						--a[2];
						if(a[1]>=5)a[1]-=5;
						else a[1]=0;
					}
					else{
						a[2]=0;
						if(a[1]>=9)a[1]-=9;
						else a[1]=0;
					}
				}
				if(a[3]==2){
					if(a[2]>=3){
						a[2]-=3;
						if(a[1]>=6)a[1]-=6;
						else a[1]=0;
					}
					else{
						int i=18-4*a[2];
						a[2]=0;
						if(a[1]>=i)a[1]-=i;
						else a[1]=0;
					}
				}
				if(a[3]==1){
					if(a[2]>=5){
						a[2]-=5;
						if(a[1]>=7)a[1]-=7;
						else a[1]=0;
					}
					else{
						int i=27-4*a[2];
						if(a[1]>=i)a[1]-=i;
						else a[1]=0;
						a[2]=0;
					}
				}
				a[3]=0;
			}
		}
		while(a[2]>0){
			++ans;
			if(a[2]>=9)a[2]-=9;
			else{
				int i=36-4*a[2];
				if(a[1]>=i)a[1]-=i;
				else a[1]=0;
				a[2]=0;
			}
		}
		while(a[1]>0){
			++ans;
			if(a[1]>=36)a[1]-=36;
			else a[1]=0;
		}
		printf("%d\n",ans);
	}
}
```
