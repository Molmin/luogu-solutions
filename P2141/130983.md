```cpp
#include<iostream>
using namespace std;
int a[101]={0};//定义一个整数数组，用来存放集合中的正整数
bool b[101]={0};//定义一个布尔数组，用来判断该数是否符合条件
int main(){
	int n//集合·中给出的正整数个数
    int q=0;//q为符合条件的正整数的个数，初始化为0
	cin>>n;//输入n
	for(int i=1;i<=n;i++)
    	cin>>a[i];//以数组的形式输入所有正整数
	for(int i=1;i<=n;i++)
    {
		for(int j=1;j<=n;j++)
        {
			for(int k=1;k<=n;k++)
            {
				if(a[j]+a[k]==a[i]&&j!=k)//判断a[i]是否s是等于集合中另外两个（不同的）数之和的正整数
//ps:因为该集合中的元素都为正整数，所以a[i]加集合中的任意一个数都不可能等于a[i]，所以i不可能等于j和k,故不用考虑i!=j和i!=k
				{
                	b[i]=true;//如果有（至少）一对元素相加等于该数，则该数符合条件
                }
			}
		}
	}//来自大括号的支配（逃
	for(int i=1;i<=n;i++){q+=(int)b[i];}//如符合条件则总数+1(即转换为int的true)	
	cout<<q;//输出总数
	return 0;//完事
}
```
