#### 这道题一看就想用模拟解决，直接上代码吧（解释都写在代码里）
```
#include<iostream>
using namespace std;
int n,m,k,i,a[1000001],b,c,s;//数组a用来记录踩到的次数，c表示踩到的地方
int main(){
	cin>>n>>m>>k;
	for(i=1;i<=m;i++){
		cin>>b;
		c=b;
		while(a[c]<2){//当没有第二只兔子踩这里
			a[c]++;//标记已经踩过
			c+=k;//跳到下一个地方
			c%=n;//以防超出范围
		}
	}
	for(i=0;i<=n-1;i++){//检查每一个格子踩了几次
		if(a[i]==0){//没被踩过
			s++;
		}
	}
	cout<<s;
} 
```
### 在判断踩没踩过时，既可以使用整型标记，又可以使用布尔型（bool）判断。
