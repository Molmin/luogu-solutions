#include<iostream>
#include<string>
using namespace std;
string a,s[4700],d[4700];
char X[]={'0','2','2','2','3','3','3','4','4','4','5','5','5','6','6','6','7','0','7','7','8','8','8','9','9','9','0'};//预处理
bool v;
int main()
{
	cin>>a;
	for(int i=0;i<=4616;i++)
		cin>>s[i];
	for(int i=0;i<=4616;i++)
	{
		d[i]=s[i];
		for(int j=0;j<=s[i].size()-1;j++)
		{
			s[i][j]=X[s[i][j]-'A'+1];//翻译
		}
	}
	for(int i=0;i<=4616;i++)
	{
		if(s[i]==a)
		{
			cout<<d[i]<<endl;
			v=1;//标记
		}
	}
	if(!v)
	cout<<"NONE";//输出反例
}
