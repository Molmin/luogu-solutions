```cpp
#include<cstdio>
#include<iostream>
using namespace std;
char a[10000];
int main(){
	int n,len=1,sum=0,k=0;
	scanf("%d",&n);getchar();
	a[1]=' ';
	for(int i=1;i<=n;i++){
		while(true){
			a[++len]=getchar();
			if(a[len]=='\n') len--;
		//	cout<<"|"<<a[len-1];
			if(a[len]=='.'||a[len]=='!'||a[len]=='?'){
				//for(int i=1;i<=len;i++)cout<<a[i];
				len=1;a[1]=' ';
				break;	
			}
			
			if(a[len-1]==' '&&a[len]>='A'&&a[len]<='Z'){
				while(true){
					//cout<<a[len]<<";";
					a[++len]=getchar();
					if(a[len]==' '||a[len]=='.'||a[len]=='?'||a[len]=='!'){
	
						sum++;
						k=1;
						break;
					}
					if(a[len]<='9')break;
 				}
			}
 			if(k==1)
			 	if(a[len]=='.'||a[len]=='?'||a[len]=='!'){
				 	k=0;
				 	len=1;a[1]=' ';break;
				}
				else k=0;
		}
		printf("%d\n",sum);sum=0;
	}
	return 0;
}
```
