又是一道字符串题.....


------------

我们先看题，

题意是只有一行，

但需要读入 _N_ 个 ' . ' 或 ' ! ' 或 ' ? '  , 

故不放心的我用了getchar，

这样只要每次遇到结束符号时输出就行了。

下面见代码吧：

c++

```cpp
#include<bits/stdc++.h>
using namespace std;
char a;//在线算法，可省空间 
int pd;//判断  是否可以开启新的名词判断 
int n;
int tp;//特判  是否符合名词 
int ans=0;//答案 
int main(){
	cin>>n;
	a=getchar();
	int i=1;
	while(i<=n){//初始化 
		pd=1;// 第一个单词就可以 
		a='0';// 特殊  不初始化将无法读入 
		ans=0;// 清零 
		tp=0;// 
		i++;// 
		while(a!='.'&&a!='!'&&a!='?'){ 
			a=getchar();
			if(a>='A'&&a<='Z'&&pd==1){//-// 是否是新单词 
				ans++;tp=1;pd=0;        // 
			}//------------------------// 

			if(tp==1){//---------------------------------------------------------------//判断是否合法 
				if(a<='9'&&a>='0'){ans--;tp=0;}                                       // 含有数字？ 
				else if((!(a>='A'&&a<='Z'))==true&&(!(a<='z'&&a>='a'))==true)tp=0;   // 
			}//---------------------------------------------------------------------// 

			if((!(a>='A'&&a<='Z'))==true&&(!(a<='z'&&a>='a'))==true&&(!(a<='9'&&a>='0'))==true)pd=1;else pd=0;
			//判断是否结束 本次判断 
		}
		cout<<ans<<endl;
	}
	return 0; 
}
```
