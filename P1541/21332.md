题目十分简单，实际上就是背包问题

考虑到题目所给一个数据点一秒，因此n\*n\*n\*n=2560000是完全没有问题的

就是打代码麻烦了点，注意别打错！！

```cpp
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int dp[41][41][41][41]={0},n[351],x[5],y[5];
int main()
{
    int a,b;
    cin>>a>>b;
    for(int i=1;i<=a;i++)
      cin>>n[i];
    for(int i=1;i<=b;i++)
      {
          int c;cin>>c;
          x[c]++;
      }
    for(y[1]=0;y[1]<=x[1];y[1]++)
      for(y[2]=0;y[2]<=x[2];y[2]++)
        for(y[3]=0;y[3]<=x[3];y[3]++)
          for(y[4]=0;y[4]<=x[4];y[4]++)
            {
             if(y[1]>0&&dp[y[1]-1][y[2]][y[3]][y[4]]>dp[y[1]][y[2]][y[3]][y[4]])//背包的过程，枚举四种卡片
                 dp[y[1]][y[2]][y[3]][y[4]]=dp[y[1]-1][y[2]][y[3]][y[4]];
             if(y[2]>0&&dp[y[1]][y[2]-1][y[3]][y[4]]>dp[y[1]][y[2]][y[3]][y[4]])
                 dp[y[1]][y[2]][y[3]][y[4]]=dp[y[1]][y[2]-1][y[3]][y[4]];
             if(y[3]>0&&dp[y[1]][y[2]][y[3]-1][y[4]]>dp[y[1]][y[2]][y[3]][y[4]])
                 dp[y[1]][y[2]][y[3]][y[4]]=dp[y[1]][y[2]][y[3]-1][y[4]];
             if(y[4]>0&&dp[y[1]][y[2]][y[3]][y[4]-1]>dp[y[1]][y[2]][y[3]][y[4]])
                 dp[y[1]][y[2]][y[3]][y[4]]=dp[y[1]][y[2]][y[3]][y[4]-1];
             dp[y[1]][y[2]][y[3]][y[4]]+=n[1+y[1]+2*y[2]+3*y[3]+4*y[4]];
            }
    cout<<dp[x[1]][x[2]][x[3]][x[4]];
    return 0;
}
```
