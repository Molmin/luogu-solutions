#include <bits/stdc++.h>
using namespace std;
void print(int x)
{
	if(x) puts("YES");
	else puts("NO");
	exit(0);
}
int main()
{
	int n;
	cin>>n;
	if(n/10==2||n/10==7||n/10==9) print(0);
	if(n/10==1) 
	{
		if(n==12) print(1);
		else print(0);
	}
	if((n%10>1&&n%10<7)||n%10==8) print(1);
	if((n%10==0)&&((n/10>1&&n/10<7)||(n/10==8))) print(1);
	print(0);
	return 0;
}
