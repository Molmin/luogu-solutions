### 好像前面几位大佬的题解比较难以理解，我来发个比较好理解的题解吧，
# 上题解：
```
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;//输入
	if(n>m)//找出比较大的那个数
	{
		n-=m;//先将n设置为已穿过的
		cout<<m<<" "<<n/2<<endl;//输出穿不同袜子的最大次数m，再输已穿过的剩下的最大的数n整除2，就是穿相同袜子的最大次数
	}
	else//否则
	{
		m-=n;//先将m设置为已穿过的
		cout<<n<<" "<<m/2<<endl;//输出穿不同袜子的最大次数n，再输已穿过的剩下的最大的数m整除2，就是穿相同袜子的最大次数
	}
 	return 0;
}
```
