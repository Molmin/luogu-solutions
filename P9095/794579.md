### 题目思路
枚举+模拟： {1,2,3,4,5}  与 {A,B,C} 这两个集合进行全排列之后得到的每一个字符串与 Matthew 的每个想法适合的题目位置都判断一遍,再设一个数组存储这 15 个字符串在 Matthew 的想法中出现的次数,最后比较出现次数是否比要求分配的题数小,最后输出。
### 完整代码
```cpp
#include<bits/stdc++.h>
using namespace std;
int n;
string c;//用字符串存储 Matthew 的每个想法适合的题目位置
int s[16];// Matthew 的每个想法适合的题目位置的出现次数
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>c;
		if(c=="1A") s[1]++;
		if(c=="1B") s[2]++;
		if(c=="1C") s[3]++;
		if(c=="2A") s[4]++;
		if(c=="2B") s[5]++;
		if(c=="2C") s[6]++;
		if(c=="3A") s[7]++;
		if(c=="3B") s[8]++;
		if(c=="3C") s[9]++;
		if(c=="4A") s[10]++;
		if(c=="4B") s[11]++;
		if(c=="4C") s[12]++;
		if(c=="5A") s[13]++;
		if(c=="5B") s[14]++;
		if(c=="5C") s[15]++;
		/*直接把每一种可能出现的情况判断一遍*/
	}
	if(s[1]<1||s[2]<1||s[3]<1||s[4]<1||s[5]<1||s[6]<1||s[7]<1||s[8]<1||s[9]<1||s[10]<1||s[11]<1||s[12]<1||s[13]<2||s[14]<2||s[15]<2) cout<<"NIE\n";
	else cout<<"TAK\n";
	/*最后的判断：判断出现次数是否比要求分配的题数小,如果小输出 "NIE" ,否则输出 "TAK" */
	return 0;
}
```
