我们看题：

* 有 $0$ ~$9$ 的某一个数字。
* $5$ 和 $2$ 可互换， $6$ 和 $9$ 可互换。
* 用已有的数字拼出尽可能多的数字 $T$ 。

我们再看数据范围:

* 冰块少于 $200$ 个。
* $T$ 小于 $10000$ 。

所以这道题模拟时间复杂度就 $20000$ 。

模拟就过得去！

所以我们就每次摆出一个 $T$ ，

若可以接着摆，

不行就放弃，输出。

下面上代码吧：


```cpp
#include<bits/stdc++.h>//万能头很香 。 
using namespace std;
char a[17];//存 T 。 
char b;
int c[10];//存每种冰块个数 。 
int sl;//a[] 数组长度 。 
int ans=0; 
int main(){
	gets(a);
	sl=strlen(a);
	int pd;//pd 是否拼成冰块 ，是再判断 ，不是跳出 。 
	
//-----------------------------------------------//switch 打表结构 。 
	while(1){                                   // 
		b=getchar();                           // 
		if(b>'9'||b<'0')break;                // 
		switch(b){                           // 
			case '0':c[0]++;break;          // 
			case '1':c[1]++;break;         // 
			case '2':c[2]++;break;        // 2 5 可互换 ，不如都计入 2 中 ，一起结算 。 
			case '3':c[3]++;break;       // 
			case '4':c[4]++;break;      // 
			case '5':c[2]++;break;     //  
			case '6':c[6]++;break;    // 6 9 可互换 ，不如都计入 6 中 ，一起结算。 
			case '7':c[7]++;break;   // 
			case '8':c[8]++;break;  // 
			case '9':c[6]++;break; // 
		}                         // 
	}                            //
//------------------------------// 
	do{
		pd=1;
		for(int i=0;i<sl;i++){//-----------------------// 
			if(c[a[i]-'0']>0)c[a[i]-'0']--;           // 判断是否可以 ， 
			else {                                   // 
				if(a[i]=='9'&&c[6]>0)c[6]--;        // 可以的话减去 ， 
				else if(a[i]=='5'&&c[2]>0)c[2]--;  // 
				else pd=0;                        // 不行的话 pd=0 跳出循环 。 
			}                                    // 
		}//-------------------------------------// 
		if(pd==1)ans++;
	}while(pd==1);
	cout<<ans<<endl;
}
```
