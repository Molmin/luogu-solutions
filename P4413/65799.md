          end;
    end.

    begin
      readln(r1,s);

      while 1=1 do//纳尼！死循环！很令你震惊吧，但死循环也是可以对的。

              writeln(i);//和方法一一样的部分我就不说了
              exit;//退出了不就不怕死循环了吗，怎么样，信我了吧。
            end;
          i:=i+1;//记得累加，不然也许就真死循环了。
        end;
    end.
