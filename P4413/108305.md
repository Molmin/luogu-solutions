	我的化简过程：
        s=(r1+r2)/2
        s=r1/2+r2/2
		r2/2=s-r1/2
     所以s-r1/2算出来的结果再*2就可以啦！

	//上面是万能头文件，它包含了近乎c++的所有头文件

	using namespace std;

	int main(){	
		float s(0),r1,r2;
		cin>>r1>>s;
		r2=s-r1/2;
		r2*=2;
		int r=r2;
		cout<<r;
		return 0;
	}
