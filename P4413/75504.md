## 标程
```cpp
//S=(R1+R2)/2;
#include<iostream>
using namespace std;
int main(){
	int R1,R2,S;
	cin>>R1>>S;
	for(R2=-2020;R2<=2020;++R2){
		if((R2+R1)/2==S){
			cout<<R2;
			break;
		}
	}
	return 0;
}
```
