Program P1422;
Var
  n: Integer;//用电量题中描述为整数
  Ans: Real;//价格会出现小数情况
Begin
  ReadLn(n);//读入用电量
  //函数分为三段，按情况计算
  If n <= 150 Then Ans := n * 0.4463;//150度以下按第一梯度价格收费
  If (150 < n) And (n <= 400) Then Ans := 150 * 0.4463 + (n - 150) * 0.4663;//150度按第一梯度收费，余下部分按第二梯度收费
  If 400 < n Then Ans := 150 * 0.4463 + 250 * 0.4663 + (n - 400) * 0.5663;//150度按第一梯度收费，(400-150)度按第二梯度收费，超出400度按第三梯度收费

  WriteLn(Ans:0:1);//输出保留一位小数
End.
