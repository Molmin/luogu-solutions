var
  n:real;
  m:longint;
begin
  n:=0;
  readln(m);
   case m of
    0..150:n:=0.4463*m;       //0-150千瓦时的部分收费情况
    151..400:n:=0.4463*150+0.4663*(m-150);    //151-400收费情况，150以下钱+150以上
    401..100000:n:=0.4463*150+0.4663*250+0.5663*(m-400);   //同上，401及其以上收费部分
  end;
  writeln(n:0:1);     //保留一位小数，场宽0
end.
