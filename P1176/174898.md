因为本题只能向下或向右走，所以递推式可以轻松得到a[i][j]=a[i-1][j]+a[i][j-1]

------------

本题特殊之处在于有m处障碍，所以需要一个b数组标记障碍
有了这个思路后，可以~~轻松~~得解。

------------

程序实现如下：
```cpp
#include<iostream>
using namespace std;
int a[1001][1001]={0},b[1001][1001]={0},n,m,x,y;
int main(){
	cin>>n>>m;
	for(int i=1;i<=m;i++){
		cin>>x>>y;
		b[x][y]=1;
	}
	a[1][1]=1;//起始位置为1
	for(int i=1;i<=n;i++){
		for(int j=1;j<=n;j++){
			if(b[i][j]==1){
				a[i][j]=0;//障碍置为0
			}else if(i!=1||j!=1){
				a[i][j]=(a[i-1][j]%100003)+(a[i][j-1]%100003))%100003;//孙子同余定理，**不然会炸掉**
			}
		}
	}
	cout<<a[n][n];
	return 0;
}
```
