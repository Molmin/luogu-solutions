这题有很多坑点，比如整点，零几点的补位等...

这里的除法有些特殊，因为$100/99$程序结果会为$1$，所以需要特判，如果还有余数的话，那就$+1$：
```cpp
if(s%v==0) cnt=s/v;
else cnt=s/v+1;
```
特判整点方法，这里需要一个$sum$来计算需要的**天**数，为防止超过24小时，所以要%$24$，重点是要$+1$，是因为整点程序会把$12:00$自动判断为$11:60$，所以在把分钟归0的同时把小时数$+1$：
```cpp
if(h%8==0) sum=h/8;
else sum=h/8+1;
cout<<((8+sum*24)-h)%24+1<<":00";
```
零几点零几分等也很好判断，这里要分情况，看你输出的分钟是什么，那就特判分钟是否小于$10$，如果是，就多输出一个$0$，不是的话就正常输出：
```cpp
if(60-(cnt%60)<10) cout<<((8+sum*24)-h)%24<<":0"<<60-(cnt%60);
else cout<<((8+sum*24)-h)%24<<":"<<60-(cnt%60);
```
现在坑点都解决的差不多了，全程$O(1)$，挂下代码，里面有注释辅助理解。

$Code:$
```cpp
#include<bits/stdc++.h>//万能头文件
using namespace std;
int s,v,cnt,h,sum;//定义，cnt为分钟总数，h是小时数，sum是计算天数（24小时）代替的
int main()
{
	cin>>s>>v;//输入
	if(s%v==0) cnt=s/v;//除法处理
	else cnt=s/v+1;
	cnt+=10;//根据题目要求垃圾分类+10
	if(cnt<=60)//特判1，一个小时内的情况
	{
		if(60-cnt<10) cout<<"07:0"<<60-cnt;
		else cout<<"07:"<<60-cnt;
	}
	else
	{
		if(cnt%60==0) h=cnt/60;//小时数+除法处理
		else h=cnt/60+1;
		if((8-h)<0)//继续特判，超过8个小时
		{
			if(h%8==0) sum=h/8;
			else sum=h/8+1;
			if(60-cnt%60==60) cout<<((8+sum*24)-h)%24+1<<":00";//判断整点
			else
			{
				if(60-(cnt%60)<10) cout<<((8+sum*24)-h)%24<<":0"<<60-(cnt%60);//判断小于10分钟的情况
				else cout<<((8+sum*24)-h)%24<<":"<<60-(cnt%60);
			}
		}
		else//否则，就是8个小时以内的情况
		{
			if(60-(cnt%60)<10) cout<<"0"<<8-h<<":0"<<60-(cnt%60);
			else cout<<"0"<<8-h<<":"<<60-(cnt%60);
		}
	}
	return 0;//程序结束，返回0
}
```
