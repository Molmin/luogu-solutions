__先来说说这题的整体思路:__

__①定义并输入变量，n和计数变量ans；__

__②核心部分，记得加上特判n==0；__
```
一、若n=0,直接输出1，终止程序；

二、若n!=0，while循环，每次/16，除到n<=0为止；
```
__③在核心部分中判断，注意加上字母的坑；__

④提交！

```
#include <bits/stdc++.h>//万能头文件
using namespace std;
int ans;//定义计数变量
int main()
{
    int n;
    cin>>n;//输入
    if(n==0)//特判n=0
    {
    	cout<<1<<endl;
    	exit(0);//输出并终止程序
	}
	while(n>0)//while核心循环
	{
		int s=n%16;
		if(s==0||s==4||s==6||s==9||s==10||s==13)
			ans++;//判断只有一个圈
		else if(s==8||s==11)
			ans+=2;//判断有两个圈
		n/=16;//注意一定要加上n/16
	}
	cout<<ans<<endl;//输出
    return 0;
}
```
