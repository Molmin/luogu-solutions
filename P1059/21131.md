//见到没人用桶排序，我就来水一题
#include <iostream>
#include <cstring>
using namespace std;
int main(){
    int n,a[1009]={0},t,s=0;
    cin>>n;
    for (int i=1;i<=n;++i){
        cin>>t;
        a[t]++;//出现什么数字，就在数字所在位置加上1，代表有了
    }
    //下面的两组循环千万不能写在一起，不然会出现输出好多个s的情况，这个简单
    for (int i=1;i<=1000;++i){
        if (a[i]!=0){//如果出现了那个数字，那肯定是大于0的，这儿也可以写出>0
            ++s;//出现了，加上
        }
    }
    cout<<s<<endl;//输出有几个位置是有的，有的加1
    for (int i=1;i<=1000;++i){
        if (a[i]!=0){//如果出现了输出
            cout<<i<<" ";//别忘了空格
        }
    }
    cout<<endl;//为了美观
    return 0;
}
//桶排序的优势就是在所在位置出现了，有的话加上1，如果碰到同样的了，还是那个桶(位置)，顶多a[i]>=1了，但是由于一开始的桶数就没有重复的，因此不肯能重复
//啦啦啦，虽然桶排序浪费空间，但是也有很大优势哦:-)
//效率: 6ms
