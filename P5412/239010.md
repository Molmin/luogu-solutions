先说一下思想，开四个数组，一个存输入的所有人的身高，一个存输入的每个人的性别，另外两个分别存男生和女生的身高。

再把男生身高和女生身高分别排序，最后输出即可。

```cpp
int n,t,cnt=0,cnt1=0,zuida;//t表示有几组数据，n表示每组数据有多少人，cnt和cnt1来计算男女生各有多少人，zuida在后面才会涉及，先往下看
float a[10005];//这是用来存输入的班级所有人的身高，因为只有一位小数，所以float定义即可
int b[10005];//用来存班级所有人的性别，因为只是0或1，所以int就行了
float nan1[10005],nv[10005];//nan1数组是男生的身高，因为nan可能在c++里有定义，所以改成nan1，nv数组表示女生身高
```
上面的代码很好理解吧，把需要的东西先定义为全局变量。

对吧，输入到这里就结束了，然后让我们思考一下怎么进行分组，当然这个思考过程应该在30秒内想出来。

这时候cnt和cnt1就有用了，分别判断一下男女生人数，一边判断一边赋值。

康康这个过程的代码。

```cpp
               for(int j=0;j<n;j++)//有n个人就跑n次呗
```

### 现在来一篇完整代码，~~但是我知道各位神犇都不需要我这小蒟蒻的代码！~~

```cpp
#include<bits/stdc++.h>
using namespace std;
int n,t,cnt=0,cnt1=0,zuida;
float a[10005];
int b[10005];
float nan1[10005],nv[10005];
int main()
{
    cin>>t;
	for(int i=1;i<=t;i++)
	{
		cin>>n;
		for(int j=0;j<n;j++)
		{
			cin>>b[j];
	    }
	    for(int j=0;j<n;j++)
	    {
	    	cin>>a[j];
		}
		for(int j=0;j<n;j++)
		{
			if(b[j]==0)
			{
			    nv[cnt]=a[j];
			    cnt++;
		    }
		    if(b[j]==1)
			{
				nan1[cnt1]=a[j];
				cnt1++;
			}	
		}
		sort(nan1,nan1+cnt1);
		sort(nv,nv+cnt);
			for(int i=0;i<cnt;i++)
			{
				if(i!=cnt-1)
				cout<<nv[i]<<" ";
				else
				cout<<nv[i]<<endl;
			}
			for(int i=0;i<cnt1;i++)
			{
				cout<<nan1[i]<<" ";
			}
		zuida=max(cnt,cnt1);
		for(int i=0;i<=zuida;i++)
		{
			nan1[i]=0;
			nv[i]=0;
		}
		cnt=0;
		cnt1=0;
	}
	return 0; 
}
```
