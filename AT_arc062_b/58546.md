这道题还算简单，几个$if$解决了。

------------
思路：我们可以一直出布，但是出的石头一定不能比布少。所以我们要把布出在对方出石头的时候，而把石头出在不能出布的时候。就是尽量不要让自己输。

------------
代码：
```cpp
#include<bits/stdc++.h>//头文件
using namespace std;
string a;//出拳方式用string记录
int g,p,w;//用于记录出拳，出布和分数
int main(){
    cin>>a;
    for(int i=0;i<a.length();i++){
        if(a[i]=='g'){//如果对方出石头
            if(g>=(p+1)){//先判断能不能出布
                w++;//如果能就把分数加一
                p++;//而且把出布的次数加一
            }
            else g++;//否则平局
        }
        else{//如果出布
            if(g>=(p+1))p++;//判断能不能出布，如果能，平局
            else{//不能出的话
                g++;//出拳的次数加一
                w--;//分数减一
            }
        }
    }
    cout<<w<<endl;//最后输出分数
    return 0;//结束
}
```
