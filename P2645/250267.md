#include<bits/stdc++.h>//万能头
using namespace std;
int main(){
	int n,m,k,l,cnt1=0,cnt2=0;//命名
	cin>>n>>m;//输入两次打的个数
	for(int i=1;i<=n;i++){
		cin>>k;//输入第一次打的
		cnt1+=k;//计数器加相应的分数
		if(k==0)//特判犯规
		cnt2+=4;//给对手加分
	}
	for(int i=1;i<=m;i++){//同上
		cin>>l;
		cnt2+=l;
		if(l==0)
		cnt1+=4;
	}
	cout<<cnt1<<" "<<cnt2;//输出计数器
	return 0;//完美结束
}
