### 这道题数据范围这么小,完全可以打表
先写一个暴搜辅助打表:
```cpp
#include <iostream>
#include <cstdio>

using namespace std;

int ans=0;

void solve(int x,int y,int n) //暴搜
{
	if(!y)
	{
		if(x==1) ++ans;
		return;
	}
	if(x==1) //特判
	{
		solve(n,y-1,n);
		solve(2,y-1,n);
	}
	else
		if(x==n) //特判
		{
			solve(n-1,y-1,n);
			solve(1,y-1,n);
		}
	else
	{
		solve(x-1,y-1,n);
		solve(x+1,y-1,n);
	}
}

int main()
{
	//注意数据范围
	for(int i=3; i<=30; i++)
		for(int j=1; j<=30; j++)
		{
			ans=0;
			solve(1,j,i);
			printf("	if(n==%d&&m==%d)\n", i,j);
			printf("		cout<<%d<<endl;\n", ans);
		}
		
	return 0;
}
```
