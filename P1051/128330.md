**我用的是结构体和一个判断序号的函数（题目说奖学金一样的话需要序号小的在前）**
```c
struct student{//结构体 
	string xm;//姓名 
	int cj,py;//成绩和评议 
	char gb,xb;//干部和西部 
	int lw;//论文 
	int jxj;//奖学金数 
	int num;//序号 
}a[101];//定义数组 
```
**然后为了方便和空间不耗费太多，将学生个数和计算总奖学金的sum定义在外边**
```
int n,sum=0;//n为学生数量，sum计算全部学生的总数量 
```

定义好之后，就可以写函数了，需要用**布尔(bool）**类型
```
bool cmp(student x, student y)//cmp函数用来判断谁的序号大，序号小在前 
{
    if(x.jxj==y.jxj)return x.num<y.num;//如果奖学金一样，则序号小的在前 
    return x.jxj>y.jxj;//如果不一样，返回值 
}
```
最后开始写主程序
```c
int main()
{
	scanf("%d",&n);//数学学生数量 
	for(int i=1;i<=n;i++)//循环输入和判断 
	{
		cin>>a[i].xm>>a[i].cj>>a[i].py>>a[i].gb>>a[i].xb>>a[i].lw;//输入 
		if(a[i].cj>80&&a[i].lw>=1)a[i].jxj+=8000;//院士奖学金判断 
		if(a[i].cj>85&&a[i].py>80)a[i].jxj+=4000;//五四奖学金 
		if(a[i].cj>90)a[i].jxj+=2000;// 成绩优秀 
		if(a[i].cj>85&&a[i].xb=='Y')a[i].jxj+=1000;//西部讲学 
		if(a[i].py>80&&a[i].gb=='Y')a[i].jxj+=850;// 班级贡献 
		a[i].num=i;//记录序号 
		sum+=a[i].jxj;//计算全部学生奖学金总数	
	}
	sort(a+1,a+n+1,cmp);//排序 
	cout<<a[1].xm<<endl<<a[1].jxj<<endl<<sum;//输出 
	return 0;
}
```
