var
n,i,k,m,s,l,mm:longint;//mm是摄像头，专门监视他吸了几支烟
begin
read(n,k);//买了多少烟，几个烟蒂换一个
m:=n;
mm:=n;
l:=0;
while m>=k do begin//直到烟蒂不足
mm:=mm+m div k;//烟蒂换了烟
m:=(m div k)+l;//烟烧成了烟蒂
l:=m mod k;//三以外的烟蒂
end;
write(mm);//妻子：让你吸烟，被我逮到了吧，让我看看你吸了多少烟！！！
