```
#include<bits/stdc++.h>//万能头文件了解一下
using namespace std;//名字空间
int main()//主函数
{
    int n,a,c,d,y;//我们会用到的变量
    cin>>a>>y>>c>>d>>n;//按题意输入，这里可以分开来写，虽然可以用空格间隔输入，但还是按照题目吧
    if(n>c)y+=(n-c)*d;//这里的判断其实就是按照题目条件直接写
    if(n*a<=y)cout<<n*a;
    else cout<<y;
    cout<<endl;//换行
    return 0;//注意注意这句话不加的话比赛时对你没有任何好处
}
```
