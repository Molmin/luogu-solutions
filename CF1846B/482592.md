**题目大意**

给定一个由 `+`、`X` 和 `O` 组成的 $3\times 3$ 的字符数组，问 `+`、`X`、`O` 中是否有哪一个满足它有连续三个横向、竖向或对角线方向。

**解题思路**

注意到连续三个横向、竖向或对角线方向一定包含 $a_{0,0}$、$a_{1,1}$、$a_{2,2}$ 中的一个，所以仅需枚举即可。

**AC 代码，请勿抄袭**

```cpp
#include<bits/stdc++.h>
using namespace std;
char a[4][4];
int main() {
  ios::sync_with_stdio(0);
  cin.tie(0), cout.tie(0);
  int t;
  cin >> t;
  while(t--) {
    for(int i = 0; i < 3; i++) {
      for(int j = 0; j < 3; j++) {
        cin >> a[i][j];
      }
    }
    if(((a[0][0] == a[0][1] && a[0][1] == a[0][2]) || (a[0][0] == a[1][0] && a[1][0] == a[2][0])) && a[0][0] != '.') {
      cout << a[0][0] << '\n';
      continue;
    }
    if(((a[0][2] == a[1][1] && a[1][1] == a[2][0]) || (a[0][0] == a[1][1] && a[1][1] == a[2][2]) || (a[0][1] == a[1][1] && a[2][1] == a[1][1]) || (a[1][0] == a[1][1] && a[1][1] == a[1][2])) && a[1][1] != '.') {
      cout << a[1][1] << '\n';
      continue;
    }
    if(((a[2][2] == a[2][1] && a[2][1] == a[2][0]) || (a[2][2] == a[1][2] && a[1][2] == a[0][2])) && a[2][2] != '.') {
      cout << a[2][2] << '\n';
      continue;
    }
    cout << "DRAW\n";
  }
  return 0;
}
```
