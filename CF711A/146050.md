其实这一题是可以边输入边进行操作的 不过CF的题时间限制的很松 主要要判断的是之前改没改过字母 并且要把字母保存下来以便后期输出 其实就没什么难的

我写的是一种一边输入一边进行操作的算法，供大家欣赏
```cpp
#include<iostream>
using namespace std;
int main()
{
	int n,f=0;
	cin>>n;
	char a[n*5+1];//尽量多定义一个长度，并没有用二维
	for(int i=1;i<=n;i++)
	{
		cin>>a[(i-1)*5]>>a[(i-1)*5+1]>>a[(i-1)*5+2]>>a[(i-1)*5+3]>>a[(i-1)*5+4] ;//这个其实用好复制粘贴写起来还是蛮容易的
		if(a[(i-1)*5+1]=='O'&&a[(i-1)*5]=='O'&&!f)//以前改过就不能改了！！！
		{
			f++;//标记改过
			a[(i-1)*5+1]='+',a[(i-1)*5]='+';
		}
		if(a[(i-1)*5+3]=='O'&&a[(i-1)*5+4]=='O'&&!f)
		{
			f++;//同上
			a[(i-1)*5+3]='+',a[(i-1)*5+4]='+';
		}
	}
	if(!f)cout<<"NO\n";
	else
	{
		cout<<"YES";
		for(int i=0;i<n*5;i++)
		{
			if(i%5==0)cout<<endl;//每五个换一次行 注意：因为最开始就会换一次行，所以cout<<"YES"并不用进行换行操作
			cout<<a[i];
		}
	}
	return 0;
}
```
