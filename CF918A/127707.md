这道题数据量较小,所以可以用桶去做(用桶去记录每一个数是否是斐波那契数列上的数),具体思路:

1,把斐波那契数列求出来.

2,用桶去记录每一个数是否是斐波那契数列上的数.

3,如果是输出'O',不是输出'o'.

代码如下:
***
```cpp
#include <iostream>
using namespace std;//always
int main()//主函数
{
	int f[35],n,b[1005]={0};//开数组,F表示斐波那契上每个数的值,B是用来存储的桶
	cin>>n;
	f[0]=1;//赋初值
	f[1]=b[1]=1;//题面里已说
	for(int i=2;i<=15;i++)
	{
		f[i]=f[i-1]+f[i-2];//把斐波那契数列求出来
		b[f[i]]++;//统计一下
	}
	for(int i=1;i<=n;i++)
	{
		if(b[i]>=1)//如果是斐波那契数列上的输出'O'
		cout<<"O";
		else//不是输出'o'
		cout<<"o";
	}
	cout<<endl;//换行
	return 0;//华华丽丽的结束	
} 
```
