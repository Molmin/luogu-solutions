using namespace std;//命名空间 
int main(){//主函数 
	long long int a,b,c;//定义不多说 
	cin>>a>>b>>c;//输入也不多说 
	if(a>b+c){//判断是否赞大于踩 
		cout<<'+';//如是则输出+ 
	}//之所以要+c是因为避免有？的情况 
	else if(b>a+c){//判断是否踩大于赞 
		cout<<'-';//如是则输出- 
	}
	else if(a==b && c==0){//判断是否a=b且无不确定的干扰 
		cout<<'0';//如是则输出0 
	}
	else{//如都不是直接输出？ 
		cout<<'?';
	}

}//perfect
