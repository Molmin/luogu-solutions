同理
当m为奇数时  显而易见地  只需把 (m/2)改为（m/2+1） 即可

（可能用递归更好理解  下面的代码是用while循环实现的）

```cpp

#include <cstdio>  
#include <algorithm>  
using namespace std;  
int m,t,a[100005],i;  
int main()  
{  
    scanf("%d",&m); 
    while (m/2!=0)  
    {  
        t++;
        if (m%2==0) a[t]=m/2;
        if (m%2==1) a[t]=m/2+1;
        m/=2; 
    }
    printf("%d\n1 ",t+1);  
    for (i=t; i>=1; i--)  
        printf("%d ",a[i]); 
    return 0;
}
```
