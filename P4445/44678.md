```cpp
#include<iostream>
using namespace  std;
long long int sum;
int a[100000],n,i;//全局变量无所谓了，个人习惯
int main()
{
cin>>n;
cin>>a[0];//先输入一个，以便下面直接比较大小
for(i=1;i<n;i++)
{cin>>a[i];sum+=max(a[i],a[i-1]);}//输入同时计算
cout<<sum;
return 0;
}
```
