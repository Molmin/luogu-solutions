首先，重点是2，要考虑2是用在256上，还是用在32上。我们来看256要比32大许多，256和32只公用2，想都不用想，肯定256划算。接下来，能组成多少个256，就组成多少个，剩余的组成32，就完美完成，速度超快（逃

如何计算能组成多少个256呢？

```cpp
    sum=min(k2,min(k5,k6));//找到最多能组成256的个数，如果2很多，5没有一个，是组成不了的，那就看2，5，6的个数里（k2,k5,k6)最少的一个量，就是最多能组成的256的数量
    k2-=sum;//这一定要！
    k5-=sum;//可以不要！
    k6-=sum;//可以不要！
    ans=sum*256;//加起来！
```

解决完256后，开始搞32(此处跟上差不多，同样的道理，就不多说了）
上代码：

```cpp
#include<bits/stdc++.h>
#define ll long long//将long long的定义名变成ll，这样方便，但大佬说这是代码陋习，请注意一下！
using namespace std;
ll k2,k3,k5,k6,ans=0;//赋值
int main()
{
    scanf("%lld%lld%lld%lld",&k2,&k3,&k5,&k6);//输入
    long long sum;
    sum=min(k2,min(k5,k6));//找到最多能组成256的个数
    k2-=sum;//这一定要！
    k5-=sum;//可以不要！
    k6-=sum;//可以不要！
    ans=sum*256;//加起来
    sum=min(k2,k3);//找到能组成32的个数
    ans+=sum*32;//加起来
    printf("%lld\n",ans);//输出
return 0;
}
```
