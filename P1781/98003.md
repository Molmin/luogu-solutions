代码如下 ：
```cpp
#include<bits/stdc++.h>
using namespace std;
string a,b;
int num1,num2,n;
int main(){
	cin>>n;
	cin>>a;	//这是票最多的人（暂定）的票数，先暂定是第一人。
	num1=1;	//这是编号。
	for(int i=1;i<n;i++){
		cin>>b; //输入后面的人的票数
		num2=i+1;//此人的编号
		if(b.size()>a.size()||b>a&&a.size()==b.size()){
        //判断1.如果此人的得票数的位数比最多（暂定）的大
        //2.票数位数一样，比较就直接比较。
        //string 的比较是从首位开始按字典序比较的。如果位数不同就会有问题。
        //比如10000和9999，因为9>1，所以会说9999大。所以要有长度相同的条件。
			a=b;		//找到票数更多的，更新票最多的人（暂定）的票数、编号。
			num1=num2;  
		}
	}
	cout<<num1<<endl<<a;
	return 0;
}
```
