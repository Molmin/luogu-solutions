#include<stdio.h>
int main()
{
	double x,n,ans,i,s;
	scanf("%lf%lf",&x,&n);
	s=0;
	ans=x;
	if(n>10000005.0)n=10000005.0;//卡数据，只会到10000005.0（危险言论）
	for(i=1;i<=n;i++)
	{
  		s+=ans/i;
		ans*=x;//同上
     		
	}
	printf("%.4lf\n",s);
    return 0;
}
}
