发现，如果%4==0，此多边形至少有一对轴是平行的

下面是代码：
--

```
#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;//t组数据
	cin>>t;//输入
	while(t--){//循环
		int n;//n
		cin>>n;//输入n
		if(n%4==0)cout<<"YES"<<endl;else cout<<"NO"<<endl;//如果%4==0，那就输出YES，否则输出NO
	}
    return 0;
}
```
