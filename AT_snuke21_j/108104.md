    这道题很简单，只要判断是否相同，则输出H，反之输出D
    

------------

    #include<bits/stdc++.h>   
    using namespace std;         
    char a,b;              
    int main(){            
        cin>>a>>b;              
        if(a==b)cout<<"H"<<endl;                
        else cout<<"D"<<endl;               
        return 0;                         
    }
