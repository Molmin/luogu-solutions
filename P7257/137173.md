# AC代码（大佬勿喷）
```cpp
#include<bits/stdc++.h>//为了代码简便，就先使用万能库啦 
using namespace std;//命名空间 
int m,n;//开在主函数外是个好习惯 
int main()
{
	scanf("%d %d",&m,&n);//比cin要快 
	m=m/100+(m%100/10)*10+(m%10)*100;//对m进行反转，分别对百位，十位，个位操作 
	n=n/100+(n%100/10)*10+(n%10)*100;//对n进行翻转，同理
	if(m>n)
	{
		cout<<m;//如果m大，就输出m 
	} 
	else
	{
		cout<<n;//如果相等或n大，就输出n 
	}
	return 0;//程序快乐地结束啦~~~ 
} 
```
