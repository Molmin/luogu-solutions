```cpp
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int a,b,c,d;
	cin>>a>>b>>c>>d;
	cout<<(a+b+c+d)/60<<endl;    //神奇的东西出现了！！！
	cout<<(a+b+c+d)%60<<endl;    //神奇的东西又出现了！！！
	return 0;
}
```
