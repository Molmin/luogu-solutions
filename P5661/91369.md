**言归正传**

T2我觉得比较容易想到结构体+队列，看到别的dalao开两个结构体，本蒟蒻参不透啊！我就开了一个。写法和楼上有点类似，又有点不一样。

 _****大致思路是，每次读入，先去掉过期地铁票。若乘地铁，直接付钱，信息入队。若乘公交车，扫一遍队列，找到符合要求的第一张票，并整理队列；若没有优惠，付钱吧。_ 
 
 贴上代码：
 ```cpp
#include<bits/stdc++.h>
using namespace std;
struct node{
	int a,pr,t;//a交通工具，pr即price价格，t时间 
}szh[100005];//结构体 
node q[100005];//结构体队列，比较方便 
int n,head,tail,ans;//队头队尾 ，队首是q[head+1],队尾是q[tail] ,非常关键 
int main()
{
	//freopen("transfer.in","r",stdin);
	//freopen("transfer.out","w",stdout);//刚考完，文件不知道有没有打错，心慌慌。。。。 
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d%d%d",&szh[i].a,&szh[i].pr,&szh[i].t); 
		while(head<tail){//从队头开始扫，当发现时间相差>45，出队，否则退出 ，注意不是<=  
			if(szh[i].t-q[head+1].t>45)
			head++;
			else break;//考试的时候break没加，无限循环，运行错误，慌得一比 ，查了半小时。。。 
		}
		if(szh[i].a==0){//乘地铁，入队，价格计入答案 
			ans+=szh[i].pr;
			q[++tail]=szh[i];
		}
		else{//公交车，分类讨论 
			if(head==tail){//如果队列为空，则价格计入答案 
				ans+=szh[i].pr;
			}
			else{//队列不为空,扫描一遍 
				int p=head;// head在扫描过程中不能变，所以定义一个指针 
				int f=0;//flag标记 
				while(p<tail){//注意不是<=
					if(q[p+1].pr>=szh[i].pr){//如果找到队列中地铁票价高于当前公交车票价，停止扫描,免费 
						for(int j=p+1;j>=head+2;j--)//j的范围需要仔细思考，模拟 
						q[j]=q[j-1];//p+1之前的都前进一个单位 
						f=1;//标记，表示已扫到符合要求
						head++;//队列长度-1 
						break;//退出循环 
					}
					else{
						p++;//再往前扫 
						continue;//进入下一次循环（现在才发现是废话） 
					}
				}
				if(f==0) ans+=szh[i].pr;// 如果没有符合要求的地铁票，乖乖掏出钱包 
			}
		}
	}
	printf("%d",ans);//Van美 
	return 0;
}
```
