 那么好了发现a[0]就是个数，而所求的值即为a[n]*n所有数的和，于是发现这只是一道模拟，直接上代码——

```pascal
var i,n,x:longint;
    sum:int64;//尽管不知道有没有用，开int64是个好习惯
begin
        read(n);
        for i:=1 to n do
        begin
                read(x);//其中的数组可以直接用一个数代替，节省空间
                sum:=sum+i*x;//sum即为所求的值
        end;
        writeln(sum);//输出
end.

```
