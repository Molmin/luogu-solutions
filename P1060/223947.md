闲话不说，掏出我的AC代码：
```cpp
#include<bits/stdc++.h>//万能头文件；
using namespace std;//操作定义或注明；
int n,m,ans=-INT_MAX;//定义全局变量；
//“n”即是总金额；
//“m”即是物品总数量；
//“ans”即是最后的总价值（每次比对刷新）；
int a[111111];//定义数组a，题中每个物品的价格；
int b[111111];//定义数组b，题中每个物品的重要性；
void jm(int dep,int v,int p){//递归函数定义；
        //函数操作中有三个变量；
        //“dep”即是深度的英文简写。表示进度；
        //“v”即是购买价格的总和统计。
        //“p”即是综合价值的总和统计。
	if(v>n)//“v>n”的判断即是购买价格不超过总金；
	{
		return;//若超过则这次运算终止。
	}
	if(dep>m)//“dep>m”的判断即是进度不超过总长；
	{
		ans=max(ans,p);//若超过则比较并刷新结果；
		return;//操作完后，终止运算。
	}
	jm(dep+1,v,p);
        //递归部分，深度进1；
	jm(dep+1,v+a[dep],p+(a[dep]*b[dep]));
        //深度进1，花销加上购价，价值加上购价乘以重要性的积；
}
int main(){
	cin>>n>>m;//输入全局变量中已定义的n，m；
	for(int i=1;i<=m;i++)//“for”循环读入数组；
	{
		cin>>a[i]>>b[i];//具体数组的读入。
	}
	jm(0,0,0);//递归初始化。
	cout<<ans<<endl;//运算后得出的结果输出。
	return 0;//程序终止
}
```
