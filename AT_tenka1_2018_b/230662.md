这道题用$for$循环$k$次，每次判断奇偶，若是奇数则轮到$a$了，偶数则轮到$b$了，然后按照题意：如果是奇数就减去一，再把自己的一半给去另外一个数字；如果是偶数，就直接把自己的一半给去另外一个数字。

代码：
```c
#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,m,k,i;
	cin>>n>>m>>k;//输入
	for(i=1;i<=k;i++)
		if(i%2==1){//判断奇偶，余1为a(n就是a)
			if(n%2==0){
				m+=n/2;
				n/=2;
				//按照题意操作
			}
			else{
				n--;
				m+=n/2;
				n/=2;
				//按照题意操作
			}
		}
		else{//同上
			if(m%2==0){
				n+=m/2;
				m/=2;
			}
			else{
				m--;
				n+=m/2;
				m/=2;
			}
		}
	cout<<n<<" "<<m;//输出
}
```
