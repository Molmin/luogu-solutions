简单的循环题，一次或两次三重循环都可过，个人用的两次，循环层数不需优化。

最后结果输出时记得是输出出现最多次数的**数字点数和**。

    
```cpp
#include<bits/stdc++.h>
using namespace std;
int main(){
    int a1,a2,a3;//三个骰子的面数。
    cin>>a1>>a2>>a3;//输入不讲。
    //在输入和处理程序之间，处理程序和输出之间加空行可以让程序调理更加清晰。
    int a[81]={0};//81是因为骰子的面数分别最大是20，20，40.加1保险起见。
    int maxn=-1,maxi;//maxn是最大值，在下面比较时会有用到。
    for(int i=1;i<=a1;i++){//骰子1循环。
        for(int j=1;j<=a2;j++){//骰子2循环。
            for(int k=1;k<=a3;k++){//骰子3循环。
                a[i+j+k]++;//骰子点数和出现的次数+1.
            }
        }
    }
        //for(int i=1;i<=81;i++)cout<<"a["<<i<<"] = "<<a[i]<<endl;////仅为测试用，输出a[i]的值
    for(int i=3;i<=a1+a2+a3;i++){//最小点数是3（1+1+1）最大点数是a1+a2+a3，所以i从最小值到最大值循环。
        if(a[i]>maxn){//如果某点数出现次数大于点数出现最大值
            maxn=a[i];//最大值更新，以便之后的比较。
            maxi=i;//答案。
        }
    }
    cout<<maxi<<endl;//输出不讲
    return 0;
}
```
