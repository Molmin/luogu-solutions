这题讲的是投骰子，那么我们能不能让电脑来投呢？

答案是可以的！

电脑不怕麻烦，它可以把骰子投个几百遍，投的次数越多，答案就会越逼近真正的

答案；

c++中的rand函数，就能够生成随机数；

比如，想要生成一个0~99之间的随机数，就可以这么写：

a=rand()%100;

如果是1~100的，再加一就可以了；

更多的细节写在代码注释里

上代码

```
#include <iostream>
#include <cstdio>
#include <ctime>
#include <cstdlib>
using namespace std;
int a[100],s1,s2,s3,x,y,z,maxn=-1,ans;
int main()
  {
  	srand((unsigned) time(NULL));//加了这句话后，每一次就可以生成不同的随机数，否则它在随机生成后就不会改变了
  	cin>>s1>>s2>>s3;
  	for(int i=1;i<=20*s1;i++)//如果s比较小，那么并不用那么多的循环，所以可以把20乘上s，来根据输入控制生成的个数
  	  {
  	  	x=rand()%s1+1;
  	  	for(int j=1;j<=20*s2;j++)
  	  	  {
  	  	  	y=rand()%s2+1;
  	  	  	for(int k=1;k<=20*s3;k++)//三重循环，不多说
  	  	  	  {
  	  	  	  	z=rand()%s3+1;
  	  	  	  	a[x+y+z]++;//记录个数，不懂的话看下其他的题解都有讲
  	  	  	  	if(a[x+y+z]>maxn)
  	  	  	  	  {
  	  	  	  	  	maxn=a[x+y+z];
  	  	  	  	  	ans=x+y+z;
				          }
  	  	  	  	 
			        }
		     }
	   }
	cout<<ans;
	return 0;
  }
  ```
