经过打表找规律我们可以发现S函数是1-9循环的
让后就可以做了
打表代码
```

const int maxn = 35;
int S[maxn];
int main(){
    int t;
    read(t);
    for(int i=1;i<=9;i++)
        S[i] = i;
    for (int i = 10; i <= 30;i++){
        int sum=0;
        int tmp=i;
        while(tmp){
            sum+=tmp%10;
            tmp /= 10;
        }
        S[i] = S[sum];
        cout << i<<" "<< S[i] << endl;
    }
}
```

```cpp
    int f = 0, c = getchar();

    while (isdigit(c))

    if (f)
        t = -t;
}

    read(t);
```                                   
