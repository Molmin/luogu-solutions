思路：循环每一个页码，累加每一个罗马数字符号的个数，最后输出即可。

实现：由于页码最大只有一百，所以我们可以判断这个页码属于哪一个区间，我们可以首先处理十位和百位（十位和百位的特殊点包括一百，九十，五十，四十），再处理个位（个位的特殊点包括九，五，四）。

代码（加注意事项）：

```cpp
#include<bits/stdc++.h>
using namespace std;
int n,I,V,X,L,C; // 分别存储每个字符的数量
int main()
{
	while(cin>>n&&n)
	{
		I=V=X=L=C=0;
		for(int k=1;k<=n;k++)
		{
			int m=k;
			if(m==100) C++,m=0;
			if(m>=90) C++,X++,m-=90;
			if(m>=50) L++,m-=50;
			if(m>=40) L++,X++,m-=40;
			X+=m/10; // 以上为处理十位和百位，这里是转移到个位，以下是处理个位
			m%=10;
			if(m==9) X++,I++,m=0;
			if(m>=5) V++,m-=5;
			if(m==4) V++,I++,m=0;
			I+=m;
		}
		cout<<n<<": "<<I<<" i, "<<V<<" v, "<<X<<" x, "<<L<<" l, "<<C<<" c"<<endl; // 注意输出格式
	}
	return 0;
}
```
