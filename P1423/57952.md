var
i:integer;//记录游了几次。
sum,m,now:real;//简洁的英文（不懂的看一看：sum是存储器，存储游了几米；m是距离；now是这次能游几米。）。
begin
read(m);//读入，不解释。
sum:=2;//第一次能游2米。
now:=2;//同上。
i:=1;//第一次，记录。
while sum<m do//判断，大于等于跳出。
begin
  now:=now*0.98;//这次的游泳长度。
  sum:=sum+now;//共游了几米。
  inc(i);//累加。
end;
write(i);//输出，应该不用解释。
end.
