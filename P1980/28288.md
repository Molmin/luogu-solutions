#include<iostream>
using namespace std;
int main()
{
    int a,b=0,i=1,n,c,f;//a是这个数 b用作计量出现次数，i用做从1到n循环的变量 c用作循环的变量 f用做判断数字中是否含有a的储存量 
    cin>>n>>a;
    while(i<=n)//枚举法，将从1到n的所有情况举例 
    {
        if (a!=0)//注意   如果没有这个判断语句在输入0的时候结果会很奇怪 
        {for (c=1;c<=1000000;c=c*10)//用f/10^x次方将a的最右一位数除去（c=1时不变）若使用for(c=0;c<=5;c=c+1) f=i/(10^c) 则容易出错 
         {
            f=i/c%10;// %10 只留下此时a的最右一位 
            if(f==a)
            b=b+1;    //符合条件 出现次数+1 
         }
        }
        else //这种就是a=0的情况 
            {
                for (c=1;c<=1000000;c=c*10)
            {
              f=i/c;//先判定 防止c的值超过i 
              if(f>0)
              {f=f%10;//因为前面已经去除了最右一位（c=1时不变）直接取模 
               if(f==0)
               b=b+1;
              };    
            }
            };
        i=i+1; //最后不要忘了给i+1进入下一次循环 
    }
    cout<<b;//最后输出 
    return 0;
}
