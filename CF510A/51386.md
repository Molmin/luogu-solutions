```cpp
#include<iostream>
using namespace std;
int main()
{
    int n,m,a[100][100];
    cin>>n>>m;
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=m;j++)
        {
            if(i%2==1)行数为单数的，全部都是蛇的身体哦！
            {
                a[i][j]=1;
            }
            else
            {
                if((j==1&&i%4==0)||(j==m&&i%4!=0))//在纵向开头%4为0的，和纵向末尾%4不等于0上面做好标记，以便以后输出出来
                {
                    a[i][j]=1;
                }
                else//否则就不标记了
                {
                    a[i][j]=0;
                }
            }
        }
    }
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=m;j++)
        {
            if(a[i][j]==1)//有标记过的才输出蛇的身体
            {
                cout<<"#";这是蛇的身体
            }
            else//否则就输出“.”
            {
                cout<<".";//输出空地
            }
        }
        cout<<endl;//输出回车换行
    }
    return 0;//结束了
}
```
总体的思想，就是把纵向开头%4为0的，和纵向末尾%4不等于0的给标记好，还有行数是单数的也给标记好，然后就可以输出了！
