于是就可以愉快地AC啦~

~~有点骚的宏定义~~

```
#include<bits/stdc++.h>//愉快的万能头
using namespace std;
#define IN(n) cin>>n
#define OUT(n) cout<<n//愉快地定义宏常量
int main(){
    long long a;//愉快地定义变量
    IN(a);//愉快地输入
    OUT(a+1),OUT(2);//愉快地输出
    putchar('\n');//愉快地加上AT题的标志
    return 0;//愉快地结束
}
```
