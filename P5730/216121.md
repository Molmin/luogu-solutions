# 简单的题目
本蒟蒻在比赛的时候做这道题时，仅剩15分钟，显然没有时间再去思考，所以本蒟蒻想到了一种比较暴力的算法

#### code:
```
#include<bits/stdc++.h>
using namespace std;
char c[6][1005];
int n,x;
string s; 
int main(){
	memset(c,'X',sizeof(c));//由于X字符在数字中比较多，先初始化数组全为X
	scanf("%d",&n);
	cin>>s;//以字符串的形式输入
	for(register int j=0;j<n;++j){//不知道是暴力还是模拟，反正就按样例打下去
		x=s[j];
		int i=j+1; 
		if(x=='0'){
			c[2][3*i-1]='.';
			c[3][3*i-1]='.';
			c[4][3*i-1]='.';
		}
		if(x=='1'){
			c[1][3*i-2]='.';
			c[2][3*i-2]='.';
			c[3][3*i-2]='.';
			c[4][3*i-2]='.';
			c[5][3*i-2]='.';
			c[1][3*i-1]='.';
			c[2][3*i-1]='.';
			c[3][3*i-1]='.';
			c[4][3*i-1]='.';
			c[5][3*i-1]='.';
		}
		if(x=='2'){
			c[2][3*i-2]='.';
			c[2][3*i-1]='.';
			c[4][3*i-1]='.';
			c[4][3*i]='.';
		}
		if(x=='3'){
			c[2][3*i-2]='.';
			c[4][3*i-2]='.';
			c[2][3*i-1]='.';
			c[4][3*i-1]='.';
		}
		if(x=='4'){
			c[1][3*i-1]='.';
			c[2][3*i-1]='.';
			c[4][3*i-2]='.';
			c[5][3*i-2]='.';
			c[4][3*i-1]='.';
			c[5][3*i-1]='.';
		}
		if(x=='5'){
			c[2][3*i-1]='.';
			c[2][3*i]='.';
			c[4][3*i-2]='.';
			c[4][3*i-1]='.';
		}
		if(x=='6'){
			c[2][3*i-1]='.';
			c[4][3*i-1]='.';
			c[2][3*i]='.';
		}
		if(x=='7'){
			c[2][3*i-2]='.';
			c[2][3*i-1]='.';
			c[3][3*i-2]='.';
			c[3][3*i-1]='.';
			c[4][3*i-2]='.';
			c[4][3*i-1]='.';
			c[5][3*i-2]='.';
			c[5][3*i-1]='.';
		}
		if(x=='8'){
			c[2][3*i-1]='.';
			c[4][3*i-1]='.';
		}
		if(x=='9'){
			c[2][3*i-1]='.';
			c[4][3*i-2]='.';
			c[4][3*i-1]='.';
		}
	}
	for(register int i=1;i<=5;++i){
		for(register int j=1;j<=3*n;++j){
			putchar(c[i][j]);//输出
			if(j%3==0&&j!=3*n)putchar('.');//输出分隔数字的.
		}
		puts("");//换行
	}
   return 0;
}	
```
