#include<bits/stdc++.h>
using namespace std;
int n;
  //c0是0的个数 c5是5的个数
int c0,c5,k;
int main(){
  //输入不解释
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
                          //强行输入
		scanf("%d",&k);
                          //只有5和0
		if(k==0) c0++;
		else c5++;
	}
    //没有0，直接输出-1
	if(c0==0){
		printf("-1\n");
		return 0;
	}
        //不够1组5，有0，输出1个
	if(c5<9){
		printf("0\n");
		return 0;
	}
  //输出
	for(int i=1;i<=c5/9;i++) printf("555555555");
	for(int i=1;i<=c0;i++) printf("0");
	printf("\n");
	return 0;
}
