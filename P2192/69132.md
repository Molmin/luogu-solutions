## 话不多说，上代码：
```cpp
#include<bits/stdc++.h>
using namespace std;
int n,a,b,c,ans;
int main() 
{
    cin>>n;
    for(int i=1;i<=n;i++)
    {
    	cin>>a;
    	if(a) b++;//判断是不是5。
    	else c++;//不是0那就是5咯。
    }
    if(c==0)//90的倍数一定是10的倍数，后面肯定有0，没有0，就输出-1。
    {
    	cout<<-1;
    	return 0;
    }
    while(b>8)//90的倍数要有9的倍数个5，在此输出。
    {
    	ans=1;//标记是否输出，后面要用。
    	b-=9;
    	cout<<"555555555";
    }
	if(!ans)//没输出过那就只输出一个0。
    {
    	cout<<0;
    	return 0;
    }
    while(c--)//否则输出所有0。
    {
    	cout<<0;
    }
    return 0;
}
```
