#include <bits/stdc++.h> //万能头万岁
using namespace std;
int main()
{
	long long n,ans=0; //n是人数，ans是结果(其实本来想用sum表示人数，不过现在大家都用ans那就跟紧潮流吧)
	cin>>n;
	char cl[n+2];//衣服颜色
	for(int i=1;i<=n;i++)
	{
		cin>>cl[i];
	}
	for(int i=1;i<=n;i++)
	{
		int j=i+1;
		if(cl[i]!=cl[j])//如果前后二人颜色不同（比如原谅绿和基佬紫？）
		{
			ans++;//答案加一
		}
	}
	ans++;//一定要加一！这是个坑
	cout<<ans<<"\n";
	return 0;
}
