先看一下题目

首先这题目就有点难以理解
>每次ta的手可能向左或向右移动一格

什么意思呢？   
就是将所有的字符在键盘上右移一格

但是
>手不能移出键盘

那为什么不用打表呢？（滑稽   
想想只有30个字母，所以我们只用3行就可以解决了啊

代码：
```
#include<bits/stdc++.h>
using namespace std;
//map映射可实现字符对字符的调用
map<char,char> l,r;//l代表当前字符往左移一位后的字符，r同理
char c;
void init()
{
	l['q']='q',r['q']='w';//手不能移出键盘
	l['w']='q',r['w']='e';
	l['e']='w',r['e']='r';
	l['r']='e',r['r']='t';
	l['t']='r',r['t']='y';
	l['y']='t',r['y']='u';
	l['u']='y',r['u']='i';
	l['i']='u',r['i']='o';
	l['o']='i',r['o']='p';
	l['p']='o',r['p']='p';//手不能移出键盘
	//
	l['a']='a',r['a']='s';//手不能移出键盘
	l['s']='a',r['s']='d';
	l['d']='s',r['d']='f';
	l['f']='d',r['f']='g';
	l['g']='f',r['g']='h';
	l['h']='g',r['h']='j';
	l['j']='h',r['j']='k';
	l['k']='j',r['k']='l';
	l['l']='k',r['l']=';';
	l[';']='l',r[';']=';';//手不能移出键盘
	//
	l['z']='z',r['z']='x';//手不能移出键盘
	l['x']='z',r['x']='c';
	l['c']='x',r['c']='v';
	l['v']='c',r['v']='b';
	l['b']='v',r['b']='n';
	l['n']='b',r['n']='b';
	l['m']='n',r['m']=',';
	l[',']='m',r[',']='.';
	l['.']=',',r['.']='/';
	l['/']='.',r['/']='/';//手不能移出键盘
	
}
/*打表用请忽略
qwertyuiop
asdfghjkl;
zxcvbnm,./
*/
int main()
{
	init();
	cin >> c;
	if(c=='L')
	{
		while(cin >> c)//不断输入知道文件结束
			cout << r[c];//因为已经向左移一位了所以调用右移的位置
		cout << endl;
	}
	else
	{
		while(cin >> c)//同上
			cout << l[c];
		cout << endl;
	}
	return 0;
}

```
