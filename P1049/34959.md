```cpp
#include<iostream>
#include<cmath>
using namespace std;
int main(){
	int v,n,i,w[100],dp[20001],j;//实际上最好把数组定义为全局
	cin>>v>>n；
	fill(dp，dp+20000,v)；//初始化数组
	int x；
	for (i=1;i<=n;i++){
		cin>>x；
		for (j=v;j>=x;j--)//注意反向，避免多次拿取
			dp[j]=min(dp[j],dp[j-x]-x)；//注意求的是剩余空间
	}
	
	cout<<dp[v]<<endl；
	return 0；
}
```
