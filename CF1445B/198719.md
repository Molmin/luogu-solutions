```cpp
#include<bits/stdc++.h>
using namespace std;
int t,a,b,c,d;
inline int read()
{
	int x=0,f=1;
	char c=getchar();
	while(c<'0' || c>'9')
	{
		if(c=='-') f=-1;
		c=getchar();
	}
	while(c>='0' && c<='9')
	{
		x=(x<<1)+(x<<3)+(c^'0');
		c=getchar();
	}
	return x*f;
}
int main()
{
 
	t=read();
	while(t--)
	{
		a=read();
		b=read();
		c=read();
		d=read();
		printf("%d\n",max(a+b,d+c));
	}
	return 0;
}
```
