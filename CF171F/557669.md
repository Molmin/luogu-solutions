这道题很简单，但需要懂得一些筛质数的方法。

### 埃氏筛法：

埃氏筛法就是不停的把数的倍数筛掉，剩下的就是质数，而且它的时间复杂度也很低，写起来也很方便,判断质数时就判断数组值就行了。

注意：要开 bool 类型的数组，不然内存可能不够。

埃氏筛法代码：

```cpp
void prime()//质数判断
{
	for(int i=2;i*i<=1000000;i++)//从一到一百万筛质数
	{
		if(a[i]==0)//如果为质数
		{
			for(int j=i*i;j<=1000000;j+=i)//筛去它的倍数
				a[j]=1;//是质数为 0，不是为 1。
		}
	} 
}
```

懂得了埃氏筛法，做对这题只需要再加上暴力了。

代码：

```cpp
#include<iostream>
#include<cstdio>//头文件
using namespace std;
bool a[1000005];//数组要开到至少一百万
void prime()//埃氏筛法
{
	for(int i=2;i*i<=1000000;i++)//范围
	{
		if(a[i]==0)//如果为质数
		{
			for(int j=i*i;j<=1000000;j+=i)//筛去它的倍数
				a[j]=1;//质数为 0，合数为 1。
		}
	} 
}
int main()
{
	prime();//先判断质数
   int n;
	cin>>n;//输入
	int ans=0;// ans 为输出的答案
	for(int i=13;i<=1000000;i+=2)//暴力
	{
		if(a[i]==0)//要在质数前提下
		{
			int k=i,l=0;// l 为反转后的数
			while(k)//反转
			{
				l=l*10+k%10; 乘十加是很经典的去反方法
				k/=10;原数除以 10。
			}
			if(!a[l]&&l!=i)//判断反转后不为原数并且是质数
				ans++;//答案加一
			if(ans==n)//满足条件时
			{
				cout<<i;//输出
				return 0;//不能忘了	 
			}
		}
	} 
}
```
