#include <bits/stdc++.h>
using namespace std;
typedef unsigned long long ull;//写着方便
ull a,b,s;
int main(void){
    cin>>a>>b;//用流读省事
    while (1){
        if (a>b)a^=b^=a^=b;//优化了的换ab值
        s+=(b/a)*a*4;
        b%=a;
        if (b==0)break;
    }
    cout<<s;//输出
}
