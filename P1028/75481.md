```cpp
#include<cstdio>
using namespace std;
int k[10010];//日常打 中国联通；
int c(int x)//**主要就是在数学上总结通项公式**
{
    k[0]=1;//**特殊情况**
    k[1]=1;//**特殊对待**
    for(int i=2;i<=x;i++)
    {
        if(i%2==0) k[i]=k[i-1]+k[i/2];
        else k[i]=k[i-1];
    }//*意思就是奇数的个数=奇数减一偶数的个数，so 可以转化为偶数算*
    return k[x];
```
