我这里用的是c++和c的优势来做的这个题：


1.首先对于读入，利用的是c++里的string类型的方便；

2.接着是预处理，将字符串转换为int类型，方便计算（当然大佬可以无视）；

3.下一步就是最重要的计算了，如果计算错误的话就尴尬了；

4.最后是输出，这里我巧妙地利用了printf的特性，输出字符。（大佬可以继续无视）


好了，废话不多说，贴代码，代码中有注释，还有不懂的同学可以看注释。




```cpp
#include<iostream>
#include<cstdio>
#include<string>
#include<cstring>
using namespace std;
int a[1005],lm[1005],da[1005],k,n,lk=1;//数组一定不能开的太小，要看数据范围，我第一次交的时候开了105的数组结果wa了
string t,m;                                             //然后我硬是看不出来，gdb之后才发现给a数组赋值时怎么跑到lm数组里了。
int main()
{
    cin>>t;                                        //读入密钥
    cin>>m;                                      //读入密文
    k=t.length();
    n=m.length();
    for(int i=0;i<k;i++)                      //将密钥转换成int类型（注意字符串是从0下标开始的）
        if(t[i]>='a'&&t[i]<='z')         //如果当前值是小写的话，就减去‘a’
            lm[i+1]=t[i]-'a';
        else                                    //否则就减去‘A’
            lm[i+1]=t[i]-'A';
    for(int i=0;i<n;i++)                     //将密文转换成int类型（注意字符串是从0下标开始的）
        if(m[i]>='a'&&m[i]<='z')     //如果当前值是小写的话，就减去‘a’
            a[i+1]=m[i]-'a';
        else
        {
            a[i+1]=m[i]-'A';          //否则就减去‘A’，注意标记大写，密钥不用管大小写
            da[i+1]=1;
        }
    for(int i=1;i<=n;i++)
    {
        a[i]=a[i]-lm[lk];                   //注意阅读题目，实际上等于：明文=密文-密钥
        if(a[i]<0)
            a[i]=26+a[i];             //超出范围就调整过来
        lk++;                               
        if(lk>k)                             //注意密钥范围
            lk=1;
    }
    for(int i=1;i<=n;i++)                //最后就可以输出了
        if(da[i])
            printf("%c",a[i]+'A');
        else
            printf("%c",a[i]+'a');
    return 0;
}
```
