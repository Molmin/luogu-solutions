题目关键：         
  1.出现‘AA’-‘ZZ’ **所有26个字母在原信息中均出现过并获得了相应的“密字”**，这种情况时输出翻译过后的原信息。
  
  2.出现存在某个（或某些）字母在原信息中没有出现，**也就是26个字母没有翻译完全**，便输出“Failed”。
  
  3。出现掌握的信息里有明显的自相矛盾或错误（违反 SS 国密码的编码规则）时，也是输出“Failed”。 
  
  例如某条信息“XYZXYZ”被翻译为“ABAABA”就违反了“**不同字母对应不同密字**”的规则。
  
  **开始讲思路：**
  
  首先输入，讲这里的原因是因为上面大佬们几片题解里，都是用其他的输入法，其实用cin就可以了，让我十分不解。
  
  在此题中，有三种选择中有两种是出“Failed”，所以我们应该先判断要不要输出它，如果不用就要输出翻译了，要想知道翻译，就必须要用个数组记录26个字母对应的翻译，后面就简单了。
  
  至于这个过程，非常简单，用个for循环，在里面加判断就行了，怎么判断，这都不会？？？（拿数组吧）
  
  **话不多说，开代码！！！**
  
    #include<bits/stdc++.h>  //强大的万能头，洛谷万岁
    using namespace std;
    string st,st1,st2;    
    int a[30],b[30],c[30];  //判断前期时要用的变量。（a=这个字母的翻译，b=这个翻译的字母，c=这个字母翻译过没有）
    char ch[30];     //记录每个字母相对应的翻译
    long long len;   //字符长度，如果在for语句中直接调用字符长度，每次循环都会执行一次。（浪费时间，浪费空间的东西）
    int main()
    {
        cin>>st;
        cin>>st1;
        cin>>st2;    //不懂大佬们的的输入（前面有↑↑↑）
        len=st.size();  //字符长度，定义时有说明，这里不重复了↑↑↑
        for(int i=0;i<len;i++)
        {
          if(c[st[i]-64]==1) //如果有登记过，经行特殊判断。
          {
            if(a[st[i]-64]!=st1[i]-64||b[st1[i]-64]!=st[i]-64)  //如果今次的和以前等级的不相等，输出，结束程序运行。
            {
              cout<<"Failed";  //上面有
              return 0;    	
            }	
          }  	
          if(c[st[i]-64]==0)  //如果没有登记过就······。
          {
            c[st[i]-64]=1;  //标记是否登记过
            ch[st[i]-64]=st1[i]; //记录此单词的翻译
            a[st[i]-64]=st1[i]-64; //定义时有说明，这里不重复了↑↑↑
            b[st1[i]-64]=st[i]-64; //，定义时有说明，这里不重复了↑↑↑
          }
        }
        sort(a+1,a+1+26); //因为后面的判断和它的顺序无关，而且能让后面的运算简化，所以将他从大到小排序。
        for(int i=1;i<=26;i++)
        {
          if(a[i]==0||a[i]==a[i+1]) //检查是否有翻译不齐的情况 和 翻译相等的情况。
          {
            cout<<"Failed";  //如果，输出，结束程序
            return 0;	
          }	
        }
        len=st2.size();  //直接赋值，将以前的覆盖，并未后面做准备。
        for(int i=0;i<len;i++)
        {
          cout<<ch[st2[i]-64];  //输出当前单词的翻译,无需加空格。	
        }
        return 0;
    }
