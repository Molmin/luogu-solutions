# 思路
因为单人肯定先坐单人桌，如果没有单人桌就去坐双人桌。由于这个双人桌已经被人坐了，所以其他的双人不会选那个双人桌，就需要一个数组记录有多少个被坐了但是没坐满的双人桌，而剩下的双人桌才是能坐的，否则就只能拒绝接客。

但是要注意，单人会第一时间选择空着的双人桌，所以只有双人桌都有人坐的时候才会去拼桌，而这时就不可能接待双人的客人。
# 代码
```cpp
#include<bits/stdc++.h>
using namespace std;
int n,x,y,a[200005],ans,z;
//x是单人桌
//y是双人桌
//z是做了一个人的双人桌
int main()
{
	scanf("%d%d%d",&n,&x,&y);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		if(a[i]==1)
		{
			if(x!=0) x--;//有单人桌就直接坐
			else if(z<2*y)//只要没坐满就可以一直坐双人桌，因为是一个人坐双人桌，所以可以坐2*y个人
			{
				z++;
			}
			else ans++;//没桌子就拒绝
		}
		else
		{
			if(z<y) y--;//如果有空闲的双人座就直接坐，z>y就证明所有双人座都有一人（讨厌的人）
			else ans+=2;//否则拒绝
		}
	}
	printf("%d",ans);
	return 0;
}
```
