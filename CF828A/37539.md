本题不需要什么高端东西，老老实实模拟就能AC。

开3个变量，分别存单人桌、双人桌和坐了一个人的双人桌，分别存对应桌子的剩余个数，按顺序输入来的顾客，按照题目要求安排进座位就可以了。

```cpp
#include<bits/stdc++.h>
using namespace std;
int n, a, b, hb, t, ans;
//依次为顾客组数、单人桌数、双人桌数、
//坐了一个人的双人桌数、每组顾客的数量、拒绝顾客的数量
int main(){
    cin >> n >> a >> b;
    //输入
    //这里省略了一句话
    //hb = 0;全局变量默认初始值为0
    for (int i = 0; i < n; ++i) {
        cin >> t;//输入人数
        if (t == 1) {//如果是一个人
            if (a > 0) {//优先单人桌
                --a;
            } else if (b > 0) {//然后空双人桌
                --b;
                ++hb;//双人桌坐了一个人，变成了hb
            } else if (hb > 0){//最后看有没有坐了一个人的双人桌
                --hb;
            } else {
                ++ans;//已经没座位了，拒绝顾客数+1
            }
        } else {
            if (b > 0) {//只能做双人桌，没有就只能拒绝
                --b;
            } else {
                ans += 2;//注意这里是拒绝两个顾客（一组来两个人）
            }
        }
    }
    cout << ans << endl;
    return 0;
}
```
