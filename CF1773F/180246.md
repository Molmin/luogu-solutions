首先考虑最少能打出多少平局，再进一步分析比分情况。注意到，
- 当进球数 $a$ 与失球数 $b$ 之和小于比赛场数 $n$ 时，肯定存在平局，且最少平局数为 $n-(a+b)$ 。此时，只需要使得每一场非平局的比赛比分都是 $1:0$ 或 $0:1$ ，平局的比赛比分为 $0:0$ 即可。
- 当进球数与失球数之和大于比赛场数，就可以出现没有平局的情况,即最小平局数为 $0$ 。此时需要分类讨论得出比分情况，具体解释可见代码。
- 需要特判当比赛场数 $n$ 为 $1$ 时的情况。此时比分只能是 $a:b$ 。
```cpp
#include <bits/stdc++.h>
using namespace std;
int n,a,b;
int main()
{
    cin>>n>>a>>b;
    int tot=a+b;
    if(n==1){       //特判n=1的情况
        if(a==b) cout<<"1\n"<<a<<":"<<b;
        else cout<<"0\n"<<a<<":"<<b;
        return 0;
    }
    if(n<=tot){     //当a,b之和大于n，可以不存在平局
        cout<<"0\n";
        if(a<n){    //当a小于n时，把所有进球全部写成1:0的情况，失球的前面全部写成0:1的情况，还有剩下的失球数全部放在最后一次比赛
            for(int i=1;i<=a;i++) cout<<"1:0\n";
            for(int i=1;i<=n-a-1;i++) cout<<"0:1\n";
            cout<<"0:"<<b-(n-a-1)<<"\n";
        }
        else{   //当a大于或等于n时，把进球数和上面失球数一样处理。
            for(int i=1;i<=n-2;i++) cout<<"1:0\n";
            if(b) cout<<a-(n-2)<<":0\n0:"<<b<<"\n";     //如果b大于0，a处理时需要留出最后一场比赛，使其比分为0:b。
            else cout<<"1:0\n"<<a-(n-1)<<":0\n";        //如果b为0，就不需要留失球的场次了
        }
    }
    else{       //当a,b之和小于n，就至少会有[n-(a+b)]场平局
        cout<<n-tot<<endl;
        for(int i=1;i<=a;i++) cout<<"1:0\n";
        for(int i=1;i<=b;i++) cout<<"0:1\n";
        for(int i=tot+1;i<=n;i++) cout<<"0:0\n";
    }
    return 0;
}
```
