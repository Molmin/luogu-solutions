这道题比较简单,直接说方法:

如果可以通过调整数位使原数被$60$整除,要满足如下条件:

1. 要含$0$(被$10$整除)

2. 除了这个$0$之外还要有一个偶数(被$20$整除)

3. 数字和被$3$整除(被$3$整除)

然后就做完了(~~是不是很简单~~)

上代码
```cpp
#include<iostream>
using namespace std;
int main()
{
	int n,i,j,s,t;
	bool f1,f2;//分别判断条件1,条件2
	string a;
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>a;
		f1=0;
		f2=0;
		s=0;
		t=0;
		//初始化
		for(j=0;j<a.size();j++)
		{
			if(a[j]=='0')//如果有0
			{
				f1=1;//满足条件1
				t++;//要对0计数
				if(t==2)//如果有两个0
					f2=1;//也满足条件2
			}
			if(a[j]=='2'||a[j]=='4'||a[j]=='6'||a[j]=='8')//如果有偶数
				f2=1;//满足条件2
			s+=a[j]-'0';//累加数字和
		}
		if(f1&&f2&&s%3==0)//如果三个条件均满足
			cout<<"red"<<endl;
		else//否则
			cout<<"cyan"<<endl;
	}
	return 0;//结束
}
```
