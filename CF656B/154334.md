为什么一百万？因为用这个数正好不会超时。

我就发一篇暴力
```
#include<bits/stdc++.h>
using namespace std;
#define N 25
int ans,n;
int m[N],r[N];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
		scanf("%d",&m[i]);
	for(int j=1;j<=n;++j)
		scanf("%d",&r[j]);
	for(int i=1;i<=1000000;++i)
		for(int j=1;j<=n;++j)
			if(i%m[j]==r[j]){
				++ans;
				break;
			}
	double ans2=1.0*ans/1000000;
	printf("%.6lf",ans2);
	return 0;
}
```
