题意：给定一个数 $m$，求 $m$ 与不大于这个数的 $10$ 的幂数中最大的一个数的差。

这道题很简单，我们先打出 $10$ 的幂数的表，因为 $ m \le 10^9 $，所以我们只需要打到十的十次方即可。

然后我们枚举十的幂数，如果枚举到一个十的幂数比 $m$ 大，那么答案就是 $m$ 与上一个十的幂数的差。

代码：

```cpp
#include<cstdio>

using namespace std;

long long p10[20]={1,10,100,1000,10000,100000,1000000,10000000,100000000,1000000000,10000000000};
//打表
int t;
int n;

int main(){
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		for(int i=1;;i++){//枚举
			if(p10[i]>n){
				printf("%d\n",n-p10[i-1]);
				break;
			}
		}
	}
	return 0;
}
```
