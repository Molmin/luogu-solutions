但是定睛一看：

$2≤n≤100$

那就没什么好说的了！直接排序、输出首项末项的差即可！

直接上代码！（这可能是我代码前内容最少的一次了）

```cpp
#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,a[101];
    cin>>n;
    //n为数的个数，a数组用来储存这些数
    for (int i=1; i<=n; i++)
    {
        cin>>a[i]; //输入这些数
    }
    sort(a+1,a+n+1); //排序，这样第一项就是最小值，最后一项就是最大值
    cout<<a[n]-a[1]; //输出首项末项之差即可
    return 0;
}
```
