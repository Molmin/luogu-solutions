# **刚入坑不久的蒟蒻，第一次发题解**
这题就是一道纯模拟，思路就是每次只读入一个字符，弄几个计数器，输入字符W则华华加分，输入L则对手加分，到一定分值就记录，输入E就停止。

下面上代码，亲测可以AC

##### dalao别喷，我只是瑟瑟发抖的蒟蒻
```cpp
#include<iostream>
#include<cmath> 
using namespace std;
int a1[62510],a2[62510],b1[62510],b2[62510];
//a数组计算11分制下的分值，b数组计算21分制下的分值 
//1表示华华的得分，2表示华华对手的得分 
int main(){
	ios::sync_with_stdio(false);//加速cin和cout 
    char s;//记录每次输入的字符 
    int k1=0,k2=0,l1=0,l2=0;//统计分值
    int i=0,j=0;//统计局数 
    while(cin>>s){//每次输入一个字符 
        if(s=='E') {//输入E则跳出循环并记录目前的分值 
            a1[i+1]=k1,a2[i+1]=k2,b1[j+1]=l1,b2[j+1]=l2;
            break;
        }
        if(s=='W') k1++,l1++;//输入W则华华加分，输入L则对手加分 
        if(s=='L') k2++,l2++;
        if((k1>=11&&abs(k1-k2)>=2)||(k2>=11&&abs(k1-k2)>=2)) {
            i++;//abs是绝对值函数，这里用来让分差大于等于2 
            a1[i]=k1,a2[i]=k2;
            k1=0,k2=0;//这里统计11分制的，每统计一局比赛重置计数器 
        }
        if((l1>=21&&abs(l1-l2)>=2)||(l2>=21&&abs(l1-l2)>=2)){
            j++;
            b1[j]=l1,b2[j]=l2;
            l1=0,l2=0;//这里统计21分制的 
        }	
    }
    for(int c=1;c<i+2;c++)//输出11分制的答案 
        cout<<a1[c]<<":"<<a2[c]<<endl;
    cout<<endl;//空一行 
    for(int v=1;v<j+2;v++)//输出21分制的答案 
        cout<<b1[v]<<":"<<b2[v]<<endl;	
    return 0;
} 
```
