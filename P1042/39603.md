题意：华华与他的对手打乒乓球

先用11分制的规则来计分

然后再用21分制的规则来计分（比分相同）

代码

```
#include <bits/stdc++.h>//万能文件头
using namespace std;
char s;//11分制的
string a;//再来一个21分制
long long w,l,i;
int main(){
    while(cin>>s && s != 'E')//输入，碰到E就停止
    {
        if(s == 'W') {w++;a+=s;}
        //如果是W华华加分，a+=s是为21分制做好记录，下同
        if(s == 'L') {l++;a+=s;}
        if((w >= 11 || l >= 11) && (w - l >=2 || l - w >= 2))//这个判断是乒乓球比赛的规则啦
        { 
            cout<<w<<":"<<l<<endl;//输出
            w=0;//一局比赛结束，开始下一局
            l=0;
        }
        i++;//做了几次循环，记录好
    }
        cout<<w<<":"<<l<<endl<<endl;
        //如果有多的再输出一次，下同
    w=0;l=0;//准备进入21分制
        for(int j;j<=i;j++)
        //来个for循环，因为a是string定义的
        {
        if(a[j] == 'W') w++;//前面记录好的比分再来一次
        if(a[j] == 'L') l++;
        if((w >= 21 || l >= 21) && (w - l >=2 || l - w >= 2))//21分制比赛
        {
            cout<<w<<":"<<l<<endl;//输出
            w=0;
            l=0;
        }
    }
    cout<<w<<":"<<l;
}
```
