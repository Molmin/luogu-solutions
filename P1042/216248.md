#include<bits/stdc++.h>//万能头文件
using namespace std;
char s[62599];//62599？ 25(每行字母数)*2500(行)
int main()
{
    int n=0,a1=0,a2=0;
    char a;
    while(cin>>a)
    {
        if(a!='E')
        {
            n++;
            s[n]=a;
        }
        else break;
    }

//11分制：
    for(int i=1;i<=n;i++)
    {
        if(s[i]=='W') a1++;
        else if(s[i]=='L') a2++;
        if((a1>=11&&a1>a2&&a1-a2>=2)||(a2>=11&&a2>a1&&a2-a1>=2))//注意要相差两分！
        {
            cout<<a1<<":"<<a2<<endl;
            a1=a2=0;//注意清零
        }
    }
    cout<<a1<<":"<<a2<<endl<<endl;

//21分制：
    a1=a2=0;//注意清零
    for(int i=1;i<=n;i++)
    {
        if(s[i]=='W') a1++;
        else if(s[i]=='L') a2++;
        if((a1>=21&&a1>a2&&a1-a2>=2)||(a2>=21&&a2>a1&&a2-a1>=2))
        {
            cout<<a1<<":"<<a2<<endl;
            a1=a2=0;//注意清零
        }
    }
    cout<<a1<<":"<<a2<<endl;
    return 0;
}
