这题直接大暴力，复杂度 $O(n)$

思路如下：

- 先特判如果两个电量都是1，直接爆掉
- 然后暴力枚举查找出什么时候会没有电
- 输出，直接爆掉

---

ps:这里for循环可以当 `while(1)`使用，不加第二个判断条件就好了

```cpp
#include<bits/stdc++.h>//万能头文件
using namespace std;//using不多说
int main()
{
    int a,b;//定义输入
    cin>>a>>b;//输入
    if(a==1&&b==1){//特判如果两个电量都是1，直接爆掉
        cout<<0;//输出0
        return 0;//直接爆掉
    }
    for(int i=1;;i++){//然后暴力枚举查找出什么时候会没有电
        if(a>b) a-=2,b+=1;//判断大小减掉
        else b-=2,a+=1;//同上
        if(a<=0||b<=0) {//没有点的时候输出用了多长时间
            cout<<i;
            return 0;//爆掉
        }
    }
    return 0;    //华丽结束
}
```
