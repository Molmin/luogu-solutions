```cpp
include<bits/stdc++.h>//有了万能头，妈妈再也不用担心我会CE
using namespace std;//作为蒟蒻的我还是不装逼了
int dp[1005];
int a[105],j[105];//数组开太大会有某些不好的事发生
int main()//进入主函数
{
    int n,k;//n为时间，k草药数量
    cin>>n>>k;
    for(int i=1;i<=k;i++)
    {
        cin>>a[i]>>j[i];//循环输入采每颗草的时间和草的价值
    }
    for(int i=1;i<=k;i++)
    {
        for(int l=n;l>=0;l--)//如果正着扫就成了多重背包
        {
            if(l>=a[i]) dp[l]=max(dp[l-a[i]]+j[i],dp[l]);//传说中的动态转移方程
        }
    }
    cout<<dp[n]<<endl;//输出
    return 0;//完美结束
}
```
