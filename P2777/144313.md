```cpp
#include<stdio.h>
using namespace std;
int main(){
    int n,b[300000],max=0,ans=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&b[i]);
        if(b[i]>max)
            max=b[i];
    }
    for(int j=0;j<n;j++){
        if(b[j]+n>=max+1)
            ans++;
    }
    printf("%d",ans);
    return 0;
}
```
