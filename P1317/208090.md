**在统计山峰的时候，有部分譬如2331的平台山峰没有统计到**

废话不多说，直接上代码

```cpp
#include<iostream>
#include<string>
using namespace std;
int nu[10521];
int main(void){
	int n,i,p,ans = -1;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>p;
		if(i&&p==nu[i-1]){i--;n--;continue;}
		nu[i] = p;
	}/*输入时，删掉前后高度相同的，合并为一个山峰*/
	for(i=1;i<n-1;i++)
		if(nu[i-1]<nu[i]&&nu[i]>nu[i+1])ans++;
	cout<<ans;
	return 0;
}
```
