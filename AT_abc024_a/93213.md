## 放代码！！！：
```cpp
#include<iostream>//一坨头文件
#include<fstream>
#include<algorithm>
using namespace std;
int a,b,c,k,n,m;//定义变量
int main()
{
     cin>>a>>b>>c>>k>>n>>m;//输入各种变量
     if(n+m>=k)//判断有没有优惠
     {
     	a=a-c;//儿童片价钱减去优惠钱数
     	b=b-c;//成人片价钱减去优惠钱数
     }
	 cout<<n*a+m*b<<endl;//输出
    return 0;
}
```
