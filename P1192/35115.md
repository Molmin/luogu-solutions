一道很简单的递推题，一个数组搞定，每次可以从前K个台阶上来，类似斐波纳契数列，C语言程序如下：

```cpp
#include<stdio.h>
int k,n,m,a[100002]={1};//第0阶是一开始就在的，所以定1；
int main(){
scanf("%d%d",&n,&k);//输入
for(int i=1i<=n;i++){//第0已经定完了，所以从第1开始
for(int j=1;j<=k&&(i-j)>=0;j++){//防止到负数，不能运行，测一下防
    a[i]+=a[i-j];//每次加前K个数；
}a[i]%=100003;//题目说的}
printf("%d",a[n]);//输出
return 0;}
```
