def DFS(t,s,v,p,q):
    global ans,n,m,a
    if t==m+1 and v==n:
        ans=min(ans,s)
        return
    k=m-t+1
    if k*2+s>ans or k*p**2*q+v<n or s>=ans or v>=n:
        return
    if t==1:
        for r in range(p,m-1,-1):

                DFS(t+1,s+r*r+2*r*h,v+r**2*h,r,h)
    else:
        for r in range(p-1,m-t,-1):
            for h in range(m-t+1,q):
                DFS(t+1,s+2*r*h,v+h*r**2,r,h)
DFS(1,0,0,est,est)
