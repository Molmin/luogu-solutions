#include<cstdio>
int n,a,b,p,ans=0x7fffffff;

{
	scanf("%d",&n);
	for(int i=0;i<3;i++)
    scanf("%d %d",&a,&b),//逗号隔开
    p=n,p/=a，//p代表n，p/=a相当于p=p/a，说明n中可以买多少个含有a支铅笔的包装。
    p+=p*a<n?1:0,//在某些情况下，会有剩余，所以需要特别判断（p+=p*a<n?1:0是三目运算符，建议萌新们可以去了解一下）

}
