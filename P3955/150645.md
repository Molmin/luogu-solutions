int a[10000001],b[10000001],c[10000001];//a是图书编码，b是需求码，c是需求码的长度
int main()
{
  int n,m,i,j,k,l,o,p;//n是图书馆的所有书的数量，m是读者的数量。
  cin>>n>>m;//输入
  for(i=1;i<=n;i++)
    cin>>a[i];
  for(i=1;i<=m;i++)
    cin>>c[i]>>b[i];
  for(i=1;i<=m;i++)	  
  {
    l=10000005;//l是用来存第i个读者所需图书中编码最小的那本书的编码，我随手赋了个最大值。
	for(j=1;j<=n;j++)
	{
	  o=1;//这里之所以这样做是为了后面的取余。
	  for(k=1;k<=c[i];k++)
	    o*=10;
	  if(a[j]%o==b[i]&&a[j]<l)//a[j]%o得到的是第j本图书的后o位编码，如果第j本图书的后o位编码和第i位读者的需求码相同，且比以前符合要求的图书的编码要小的话。
	    l=a[j];//就更新最小值		
	}	
	if(l!=10000005)//如果有符合第i位读者的书的话
