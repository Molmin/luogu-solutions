//运行环境：Dev C++ 5.1.0  Windows
#include <iostream>
using namespace std;
unsigned long long Count(int s)
{
	unsigned long long Ans=1,i,j1=4,j2=4,j3=2;
	for(i=1;i<s;i++)
	{
		Ans+=j1;
		j1+=j2;
		j2+=j3;
		if(j3==2)j3=1;
		else if(j3==1)j3=2;
	}
	return Ans;
}
int main()
{
	int n,s;
	cin >> n;
	for(int i=0;i<n;i++)
	{
		cin >> s;
		cout << Count(s) << endl;
	}
	return 0;
}
