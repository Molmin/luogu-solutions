```cpp
//这道题是约瑟夫系列的问题
//当然有人会直接使用模拟，但很可惜，n<=100000
//超时~\(≧▽≦)/~，所以需要使用另一种方法
//看过具体数学这本书的OIer就能顺利的求解
//对于两人轮流出列的此类问题有递推公式：
//f[1]=1;f[2n]=2f[n]+1; f[2n+1]=2f[n]-1;  f[n]表示n个人最后出圈的那个
//至于怎么推的这里就不详细解释了╮(╯▽╰)╭不好意思╮(╯﹏╰)╭ 
#include<cstdio>
int f[200000];
int main(){
    int n;
    scanf("%d",&n);
    f[1]=1;
    for(int i=1;i<=n;i++){
        f[2*i]=2*f[i]-1;
        f[2*i+1]=2*f[i]+1;
    }
    int sum=0;
    while(f[n]!=n){
        int k=f[n];
        sum+=n-k;
        n=f[n];    
    }
    sum=sum+f[n]*2;
    printf("%d",sum);
    return 0;
}
```
