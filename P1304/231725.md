首先，用筛打个表，代码如下（我比较喜欢另外写程序）：

```cpp
#include<iostream>
using namespace std;
int n,ans;
int main(){
	freopen("hh.out","w",stdout);                 //另外开一个输出文件，好复制 
	cin>>n;                               //这里自己想打多少打多少
	bool f[n+5];
	for(int i=0;i<=n;i++)f[i]=true;               //先假设都是质数 
	for(int i=2;i*i<=n;i++){
		if(f[i]){
			for(int j=2;j<=n/i;j++)f[i*j]=false;  //标记出质数的倍数，肯定不是质数 
		}
	}
	for(int i=2;i<=n;i++){
		if(f[i]){
			cout<<i<<',';                         //结尾可能多一个逗号 
			ans++;
		}
	}
	cout<<ans;                                    //输出总数，方便一点~~ 
	return 0;
}
```
然后就只用把打出来的东西拷贝到代码里就行了
