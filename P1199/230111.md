```cpp
#include<bits/stdc++.h> //万能多方便

usingnamespace std;
int a[510][510];
int ans;
int main()
{
    int n;
    cin>>n;
    for(int i=1;i<n;i++)//如图 虽然ex的n为6 但是输入为五行
        for(int j=i+1;j<=n;j++)//j=i+1还是如图，而且后面是j<=n，结合上面就好理解了
        {
           	cin>>a[i][j];
            a[j][i]=a[i][j];//对称 如说明
        }
    for(int i=1;i<=n;i++)
    {
        sort(a[i]+1,a[i]+1+n);//排序不多说 注意是升序就好了
         
        //找最大值 因为前面sort是升序排的 所以直接找倒数第一个
        if(ans>a[i][n-1]){
        	ans=ans;
		}else{
			ans=a[i][n-1];//更新
		}
    }
    cout<<1<<endl<<ans;//hua li 的 ending
    return 0;
}
```
