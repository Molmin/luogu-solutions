#include<cstring>
using namespace std;
int a[1000001],b[1000001],c[10000001],n;int d[1000001];

{

    a[0]=1;a[1]=1;
    for(int i=1;i<=n;i++)
      {
          int t=i,k=1; 
          while(t!=0) 
            {
            b[k]=t%10;
            t=t/10;
            k++;
          }
    b[0]=k-1;              //存储i为b【】，记录b【】位数为b【0】 
    //******************************************************************************************
    memset(d,0,sizeof(d));
    for (int j=0;j<=a[0];j++)
    d[j]=a[j];               //进行乘法运算

    for (int j=1;j<=b[0];j++)
    {
        x=0;
        for (int j2=1;j2<=d[0];j2++)
        {
            a[j2+j-1]=b[j]*d[j2]+a[j2+j-1]+x;
            x=a[j+j2-1]/10;
            a[j+j2-1]%=10;
        }
        a[j+d[0]]=x;
    }
    a[0]=b[0]+d[0];

    a[0]--;

      }

}
