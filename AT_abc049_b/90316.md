这就很simple啦~

我们用字符就OK了，也不算太难

上程序：

```cpp
#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int n,m;  //变量长和宽
char a[100+50];  //定好字符数组
int main()
{
    cin>>n>>m;  //输入长、宽
    for(int i=1;i<=n;i++)  //循环到输入结束
    {
    	for(int j=1;j<=m;j++)
    		cin>>a[j];  //输入字符
    	for(int k=1;k<=m;k++)  //循环到宽度
    		cout<<a[k];  //输出记好的字符
    	cout<<"\n";  //转到下一行输出
    	for(int k=1;k<=m;k++)
    		cout<<a[k];
    	cout<<"\n";  //At最后记得换行
    }
	return 0;  //完美结束
}
```
