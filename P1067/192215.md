#include<iostream>

using namespace std;

int main() {
	int n;
	cin >> n;
	int a[101];
	for (int i = 0; i < n + 1; i++) {
		cin >> a[i];
	}
	for (int i = 0; i < n - 1; i++) {
		if (a[i] > 0 && i != 0)
			cout << "+";
		if (a[i] != 0 && a[i] != 1 && a[i] != -1) {
			cout << a[i] << "x^" << n - i;
		}
		if (a[i] == 1) {
			cout << "x^" << n - i;
		}
		if (a[i] == -1) {
			cout << "-x^" << n - i;
		}
	}
	for (int i = n - 1; i < n; i++) {
		if (a[i] > 0 && i != 0)
			cout << "+";
		if (a[i] != 0 && a[i] != 1 && a[i] != -1) {
			cout << a[i] << "x";
		}
		if (a[i] == 1) {
			cout << "x";
		}
		if (a[i] == -1) {
			cout << "-x";
		}
	}
	for (int i = n; i < n + 1; i++) {
		if (a[i] > 0 && i != 0)
			cout << "+";
		if (a[i] != 0) {
			cout << a[i];
		}
	}
}

这道题感觉就像是数据结构中的一元二次方程操作的简易版，最开始做这道题的时候错了许多细节，然后我就下载的测试点，然后根据测试点的答案和我开始的代码结果进行比较，然后发现了一些需要注意的地方。

首先，我做这道题的思路分成了三部分，

第一，是先输出第一个的指数和系数到第n-2个的指数和系数，输出的指数要注意，要输出的是n-i而不是i，因为指数是从大到小输出的，而系数是从小到大输出的。并且需要注意的是，除了第一个系数是正数不需要加+号，其他位置上的正数都需要在前面加上一个+号的。而且当系数的大小为正负1时，需要注意将系数“藏起来”。

第二，是输出倒数第二个的指数和系数，根据测试点的数据结果，需要注意当时倒数第二个时的x的指数并不需要输出成x^1，而是直接输出系数x，并且同样需要考虑正负1的情况。

第三，数输出最后一项，因为只需要输出系数就行了，因此只要考虑加正号。

最后，这三部分都需要考虑系数是否为0再输出。
