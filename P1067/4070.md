```cpp

#include<cstdio>//标准输入输出头文件。 
#define p printf//为了方便用p宏替换printf。 
using namespace std;//没用的废话。 
int main(){//主函数。 
    int n,d,s=1;//声明n:次数，d:系数，s:没有输出过。 
    scanf("%d",&n);//输入次数。 
    for(int i=n;i>=0;i--){//从最高次循环到常数项。 
        scanf("%d",&d);//输入系数 
        if(!i&&s&&!d)p("0");if(!d)continue;//处理0的情况，包括只有0要输出的情况。 
        if(d<0){p("-");d=-d;}else if(!s)p("+");//如果d<0，则一定输出减号，如果d>0，则只有当输出过之后才输出加号，顺便把b变成绝对值。 
        if(d!=1||!i)p("%d",d);//如果d绝对值不是一或者是常数项，则输出数字的绝对值。 
        if(i)p("x");if(i>1)p("^%d",i);//如果不是常数项，则输出x，如果次数大于1，则输出次数。 
        s=0;}//输出过了。 
    return 0;}//返回0，表示正常运行。
```
