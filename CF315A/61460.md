```
这题显然是自己不能开自己，那就统计一下，一次就令f为1，多次就令f为2
如果f为2，无论如何都可以做到打开（即便自己不能开自己），大家想一下
其次如果f为1，只要不是自己开自己的那种都可以打开
```
------------
```
#include<stdio.h>
int n,k=0,a[1001],b[1001],f[1001];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d%d",&a[i],&b[i]);
		if(f[b[i]]==2) continue;
		f[b[i]]++;
	}
	for(int i=1;i<=n;i++) if(a[i]==b[i]&&f[a[i]]==1||f[a[i]]==0) k++;
	printf("%d",k);
}
```
