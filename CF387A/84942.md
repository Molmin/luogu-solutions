因为他只是需要用前一个时间减去后一个时间在特判一下特殊情况就可以得出答案了，主要是输入的时候可以不用字符输入，可以节省很大分量的代码，

~~看着一群大佬字符读入瑟瑟发抖~~

```cpp
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int a,b,c,d;
	scanf("%d:%d",&a,&b);//输入的时候忽略“：”
	scanf("%d:%d",&c,&d);//输入的时候忽略“：”
	if(a<c)//如果是前一天的就加上24个小时
	{
		a+=24;
	}
	else
	{
		if(b<d)
		{
			a+=24;
		}
	}
	if(b<d){
	if((a-c-1)%24<10)
	cout<<0;//不用字符串输出的
	cout<<(a-c-1)%24<<":";
	if(b+60-d<10)
	cout<<0;//不用字符串输出的，可特判是否小于10
	cout<<b+60-d;
	}
	else
	{
	if((a-c)%24<10)//一定要记得%24，不然答案会大于24
	cout<<0;
	cout<<(a-c)%24<<":";
	if(b-d<10)
	cout<<0;
	cout<<b-d;
	}
}
```
