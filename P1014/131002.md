#include <iostream>
using namespace std;
int main()
{
    int cnt,l=1,y=1,x=1;//输出如下：x/y
    cin >> cnt;
    for(int i = 1;i<cnt;i++)
    {
        if(l%2 == 1)
        {
            x--;
            y++;
            if(x == 0)
            {
                x++;
                l++;
            }
        }
        else if(l%2 == 0)
        {
            x++;
            y--;
            if(y == 0)
            {
                y++;
                l++;
            }
        }
    }
    cout << x << '/' << y;
    return 0;
}
