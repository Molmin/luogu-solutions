这是一道很简单的博弈论问题，打搜索是一个很下的策略，不是说不好，当无路可走的时候，我们可以选择这个方法，而现在我要说的是一种名叫博弈论的东西。
具体思路在程序里面：

1 ： 把日期的天数加1，例如1900.1.1变到1900.1.2

2 ： 把月份加1，例如：1900.1.1变到1900.2.1
```cpp
#include<bits/stdc++.h>
using namespace std;
int a,b,c,n;
int main()
{
    cin>>n;
    for（int i=3;i>=1;i++)
    {
        scanf("%d%d%d",&a,&b,&c);
        if((b==9&&c==30)||(b==1&&c==30)||((b+c)%2==0))//这个我们通过找规律得到，为啥是这样呢
        //我们设2001.11.4必败，2001.10.4和2001.11.3必胜，（m+d）%2==0再推，是11.2，10.3，9.3，都是必败m+d%2！=0；所以得出这个结论。
            cout<<"YES"<<endl;
        else 
            cout<<"NO"<<endl;
    }
    return 0;
}
```
