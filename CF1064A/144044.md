```
#include <algorithm>
#include <iostream>
using namespace std;
int ans,a[3];	
int main()
{
    cin>>a[0]>>a[1]>>a[2];	//读入。为什么用数组存，要sort嘛。。
    sort(a,a+3);	//排个序才好找最小值
    while(a[0]+a[1]<=a[2])	//还不满足三角形条件，继续循环
    {
        a[0]++;	//把最短的木棒添长（其实你把次短木棒添长也没事啦233）
        ans++;	//年与时驰意与日去
        sort(a,a+3);	//然后还要再排个序
    }	//如果已经符合条件了，循环全部不会执行哦
    cout<<ans;	// 输出
    return 0;	//THE END
}
```
