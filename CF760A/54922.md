一道纯模拟水题,只需模拟即可

用一个数组记录每个月的天数

用一个s记录目前是星期几

如果当前s>7,则将周数t++

最后输出t即可

上代码:
```cpp
#include<iostream>
using namespace std;
int a[12]={31,28,31,30,31,30,31,31,30,31,30,31};//记录天数的数组
int main()
{
	int m,d,i,s,t=1;//注意t要初始为1
	cin>>m>>d;
	s=d;//s初始为d
	for(i=1;i<=a[m-1];i++)//扫一遍每一天,注意是m-1
	{
		if(s>7)//如果是新的一周
		{
			t++;//周数++
			s-=7;//减去7天
		}
		s++;//每天++
	}
	cout<<t;//输出周数
	return 0;//完美结束
}
```
