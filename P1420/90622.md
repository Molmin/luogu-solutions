上代码~
```cpp
#include<iostream>
using namespace std;
int main()
{
	int n,a,x,max=0,s=0;	//这道题其实不用开数组啦。n是总数，a是输入，x存储上一次输入的数据。max是存储最长的那个连号长度，s就是存储当前连号长度。
	cin>>n;
	for(int i=0; i<n; i++)
	{
		cin>>a;
		if(x+1==a)   s++;	//如果a能和上次输入的x相连，当前连号长度就递增。
		else    s=1;	//否则就将s设为1。因为一个数本身就能看作连号长度为1。
		x=a;	//x存储本次输入的值，以备下次用。
		if(s>max)   max=s;	//max存储最大值。
	}
	cout<<max;
	return 0;
}
```

#### 特别要注意那个“s=1”和s的初始化一定要为0！
