模拟做法&&思路：

1.先把如果k为1的情况打出来，便于推导出k>1的情况。

2.然后记住它占有的位置，下面都划分好了。

3.用公式求出要输出多少行。

4.记住应该输出什么。（wpsx）

5.模拟输出

用1来试验吧！
假设k=2，因为k=2，所以每一个数的大小的长就是2+2=4，宽就是2*2+3=7。
```cpp
#include<bits/stdc++.h>
using namespace std;
string a[100005];
int x[1005],y[1005];
string stx,st,st1,st2,st3,st4,st5;
int wpsx,k;
int main()
{     
    a[1]="      -   -       -   -   -   -   -   - ";
    a[2]="   |   |   | | | |   |     | | | | | | |";
    a[3]="      -   -   -   -   -       -   -     ";
    a[4]="   | |     |   |   | | |   | | |   | | |";
    a[5]="      -   -       -   -       -   -   - ";//简称打表
    st1=a[1];st2=a[2];st3=a[3];st4=a[4];st5=a[5];
    cin>>k;
    cin>>st;
    x[1]=1;x[2]=5;x[3]=9;x[4]=13;x[5]=17;x[6]=21;x[7]=25;x[8]=29;x[9]=33;x[0]=37;
   
    y[1]=3;y[2]=7;y[3]=11;y[4]=15;y[5]=19;y[6]=23;y[7]=27;y[8]=31;y[9]=35;y[0]=39;//算出在那个表中的位置
    
    for(int qwe=1;qwe<=3+k*2;qwe++)
    {
        if(qwe==1 || qwe==2 || qwe==1+k+1 || qwe==1+k+2 || qwe==2+2*k+1) wpsx++; //算出该输出-还是|
        stx=a[wpsx];
        for(int i=0;i<st.size();i++)
    	{
    		int w=int(st[i])-48;
    		if((w==1 || w==4)&&(wpsx==1 || wpsx==5))
    		{
    		    for(int j=0;j<=k+1;j++)
    		        cout<<" ";
    		}//如果是1和4，则上面的一行和下面的一行只有空格。
     		else
    		if(w==1 && wpsx==3)
    		{
    		    	for(int j=0;j<=k+1;j++)
    		    	    cout<<" ";
    		}//如果是1，则中间的一行只有空格。
    		else
    		{
                for(int j=x[w];j<=y[w];j++)
                {
                	if((st1[j]=='-' || st3[j]=='-' || st5[j]=='-'))
                	{
                		if((wpsx==1 || wpsx==3 || wpsx==5)&&stx[j]=='-')//如果当前有要输出-时，那么按照k个输出。
                		{
                			for(int wux=1;wux<=k;wux++)
                			    cout<<"-";
                		}
                		else //如果当前不用输出-时，但是又是1、3、5行时，那么就输出等量的空格。
                		{
                			for(int wux=1;wux<=k;wux++)
                 			    cout<<" ";            			
                		}
                	}
                	else if((st1[j]!='-' && st3[j]!='-' && st5[j]!='-')&&(wpsx==1 || wpsx==3 || wpsx==5)) //如果什么都不是并且是1、3、5行时，那么就输出空格。 
                        cout<<" ";
                	else if(stx[j]=='|' &&(wpsx==2 || wpsx==4)) //如果要输出|时和2、4行时，那么就输出|。
                        cout<<'|';
                	else if((st1[j]=='-' || st3[j]=='-' || st5[j]=='-')&&(wpsx==2 || wpsx==4))//如果遇到当前列有-并且是2、4行时，也要输出等量的空格。
                    {
                        for(int wux=1;wux<=k;wux++)
                            cout<<" ";
                    } 
                    else if((st2[j]=='|' || st4[j]=='|')&&stx[j]!='|'&&(wpsx==2 || wpsx==4))//如果当前是2、4行，并且不用输出|时，那么就输出空格。
                        cout<<" ";   	
                    else if(st2[j]!='|' && st4[j]!='|' && (wpsx==1 || wpsx==3 || wpsx==5)) //如果当前列有|时，并且正在输出1、3、5行时，那么就输出一个空格。
                    cout<<" ";
                    else if((wpsx==2 || wpsx==4) && j==x[w]+1)//因为j在x[w]和y[w]的中间的话而且正在输出2、4行的时候，那么就输出等量的空格（以防万一）
                    {
                        for(int wux=1;wux<=k;wux++)
                            cout<<" ";
                    }
                    else cout<<" ";//如果还有其他情况那么就只有空格要输出了。
                }
            }
            cout<<" ";//每一个数字输出完后都要输出一个空格。
    	}
       	cout<<endl;//换行。
    }
    return 0;
}
```
