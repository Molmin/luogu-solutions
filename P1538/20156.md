#include<cstdio>
#include<cstring>
#define re register int
char s[1000],num[10][8] = {"|-| |-|","  |   |", " -|-|- ", " -|- -|", "| |-  |", "|- - -|", "|- -|-|", " -|   |", "|-|-|-|", "|-|- -|"};
int size,ls;
int main()
{
    scanf("%d%s",&size,s);ls=strlen(s);
    for(re row=1;row<=5;!(row&1)?:puts(""),row++)
        if(row&1){
            for(re i=0;i<ls;i++){
                putchar(' ');
                for(re j=0;j<size;j++)printf("%c",num[s[i]-'0'][row]);
                printf("  ");
            }
        }
        else if(row==2){
            for(re k=0;k<size;k++,puts("")){
                for(re i=0;i<ls;i++){
                    printf("%c",num[s[i]-'0'][0]);
                    for(re j=0;j<size;j++)putchar(' ');
                    printf("%c ",num[s[i]-'0'][2]);
                }
            }
        }
        else{
            for(re k=0;k<size;k++,puts("")){
                for(re i=0;i<ls;i++){
                    printf("%c",num[s[i]-'0'][4]);
                    for(re j=0;j<size;j++)putchar(' ');
                    printf("%c ",num[s[i]-'0'][6]);
                }
            }
        }
return 0;
}
