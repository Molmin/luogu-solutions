这一题十分简单，话不多说，直接暴力。  
代码：
```cpp
#include<bits/stdc++.h>//万能头文件
using namespace std;//名字命名空间
int a,b,c;//定义本题需要的三个变量
int main()//主函数
{
	cin>>a>>b>>c;//输入
	if(c<0)//先判断比零小的情况
	{
		for(int i=1;i<=abs(c);i++)//abs函数：返回一个数的绝对值。因为c是负数，所以要先把它变成正数。
		{
       	    //这里的工作原理这样子的：
            //■←■←……■←■←
            //↓→→→→→→→→↑
            
			if(b==0)b=a;//↓→→→→→→→→↑部分
            
			b--;//■←■←……■←■←部分
            
		}
	}
    //上面把b--;放在if(b==0)b=1;后是为了防止b=0的特殊情况
	if(c>0)//再判断比零大的情况
	{
		for(int i=1;i<=c;i++)//因为这里的c>0，所以就不需要abs了，节省时间。
		{
     	    //这里的工作原理这样子的：
            //■→■→……■→■→
            //↑←←←←←←←←↓
            
			if(b==a)b=1；//↑←←←←←←←←↓部分
            
			b++;//■→■→……■→■→部分
		}
	}
    //上面吧b++;放在if(b==a)b=1;后是为了防止b=a的特殊情况
    
    //为什么没有0的情况？0就不动就行了。
    
	cout<<b;//最后输出位置
	return 0;//完结撒花
}
```
