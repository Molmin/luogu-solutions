代码思路：纯模拟

以下是代码（解释都在代码里面）：

```cpp
#include<iostream>
using namespace std;
int i,j,k,n,coin;
//i：那些天应发的金币个数。j：应发i个金币的天数。k：应发天数。n：实发天数。 
int main()
{
    cin>>k;
    for(i=1;;i++)
    {
        j=0;  //记得清零。 
        while(j<i)  //发完了i天就完成，继续发之后共i+1天的金币。 
        {
            coin+=i;  //累加金币。 
            j++;  //该发i个金币的天数又过了一天。 
            n++;  //总天数又过了一天…… 
            if(n==k){  //如果 应发天数==实发天数 就完成了整个计算。 
                cout<<coin<<endl;  //输出金币总个数。 
                return 0;  //整个计算已完成且已输出，退出整个程序。 
            }
        }  //内循环一次代表一天。  
    }  //外循环完成一次代表发i个金币的日子结束。 
}
```
