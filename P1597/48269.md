```cpp
#include <cstdio>
#include <cstring>
char st[10001];//用来存储读入的字符串
int main()
{
    int a=0,b=0,c=0;//初始化a、b、c
    scanf("%s",st+1);//读入字符串
    int len=strlen(st+1);//用len来存储字符串的长度
    for(int i=2;i<=len;i++)//从2开始，因为我们等下要搜的是冒号
    {
        if(st[i]==':' && st[i+1]=='=')//如果找到了赋值变量
        {
            if(st[i-1]=='a')//如果是给a赋值
            {
                if(st[i+2]=='a')//如果":="后面的是字母a
                {
                    a=a;//那么就把a赋值给a
                }
                else if(st[i+2]=='b')//如果":="后面的是字母b
                {
                    a=b;//那么就把b赋值给a
                }
                else if(st[i+2]=='c')//如果":="后面的是字母c
                {
                    a=c;//那么就把c赋值给a
                }
                else//如果":="后面的数是数字（数字只有一位）
                {
                    a=st[i+2]-'0';//将":="后面的那个数值赋值给a
                }
            }
            else if(st[i-1]=='b')//如果是给b赋值
            {
                if(st[i+2]=='a')//如果":="后面的是字母a
                {
                    b=a;//那么就把a赋值给b
                }
                else if(st[i+2]=='b')//如果":="后面的是字母b
                {
                    b=b;//那么就把b赋值给b
                }
                else if(st[i+2]=='c')//如果":="后面的是字母c
                {
                    b=c;//那么就把c赋值给b
                }
                else//如果":="后面的数是数字（数字只有一位）
                {
                    b=st[i+2]-'0';//将":="后面的那个数值赋值给b
                }
            }
            else if(st[i-1]=='c')//如果是给c赋值
            {
                if(st[i+2]=='a')//如果":="后面的是字母a
                {
                    c=a;//那么就把a赋值给c
                }
                else if(st[i+2]=='b')//如果":="后面的是字母b
                {
                    c=b;//那么就把b赋值给c
                }
                else if(st[i+2]=='c')//如果":="后面的是字母c
                {
                    c=c;//那么就把c赋值给c
                }
                else//如果":="后面的数是数字（数字只有一位）
                {
                    c=st[i+2]-'0';//将":="后面的那个数值赋值给c
                }
            }
        }
    }
    printf("%d %d %d",a,b,c);//输出a和b和c的值
    return 0;
}
```
