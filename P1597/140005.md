#include <iostream>
using namespace std;
int main()
{
    char g,h,j,k,l;//其实只有g和k是有用的，其余都是龙套
    int a=0,b=0,c=0;//变量
    while(cin>>g>>h>>j>>k>>l)//c++博大精深
    {
        if(g=='a')//千万注意l是分号！k才是有用的！！！
        {
            if(k>='0'&&k<='9')//如果是数字
                a=k-48;
            if(k=='b')//如果是其他变量
                a=b;
            if(k=='c')
                a=c;
        }
        if(g=='b')//同上
        {
            if(k>='0'&&k<='9')
                b=k-48;
            if(k=='a')
                b=a;
            if(k=='c')
                b=c;
        }
        if(g=='c')//同上
        {
            if(k>='0'&&k<='9')
                c=k-48;
            if(k=='b')
                c=b;
            if(k=='a')
                c=a;
        }
    }
    cout<<a<<" "<<b<<" "<<c;
}
