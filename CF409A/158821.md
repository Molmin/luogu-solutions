这是一道字符串模拟题，还蛮简单的（我也只会刷这种题）

思路：

1.首先输入两个字符串a,b。

2.从0到a的长度扫一遍(如果你想用b的也可以）。

3.因为8< 代表剪刀；() 代表石头；[ ] 代表布。我们判断当前a[i]如果是 8, [  ( 的话，就是对应着 8< [ ] ()。

4.按石头剪刀布的规则，算出TEAM 1和TEAM 2分别赢了多少次，最后比较输出结果。


代码：

```
#include<iostream>
using namespace std;
string a,b;
int w1,w2;//w1表示TEAM 1胜利的次数，w2表示TEAM 2胜利的次数。
int main()
{
	cin>>a;//输入，注意两个字符串最好不要一起输。
	cin>>b;
	for(int i=0;i<a.length();i++)//字符串记得要从0开始
	{
		if(a[i]=='8')//如果当前a[i]是8，也就是8<
		{
			if(b[i]=='(') w2++;//如果当前b[i]是(，也就是(),则TEAM 2赢的次数++。
			if(b[i]=='[') w1++;//如果当前b[i]是[，也就是[],则TEAM 1赢的次数++。
		}
		if(a[i]=='(')//如果当前a[i]是(，也就是()
		{
			if(b[i]=='[') w2++;//如果当前b[i]是[，也就是[],则TEAM 2赢的次数++。
			if(b[i]=='8') w1++;//如果当前b[i]是8，也就是8<,则TEAM 1赢的次数++。
		}
		if(a[i]=='[')//如果当前a[i]是[，也就是[]
		{
			if(b[i]=='8') w2++;//如果当前b[i]是8，也就是8<,则TEAM 2赢的次数++。
			if(b[i]=='(') w1++;//如果当前b[i]是(，也就是(),则TEAM 1赢的次数++。
		}
	}
	if(w1>w2) cout<<"TEAM 1 WINS";//比较输出
	if(w1<w2) cout<<"TEAM 2 WINS";//比较输出
	if(w1==w2) cout<<"TIE";//比较输出
	return 0;
}
```
