这道题非常简单，只需一位一位的判断再替换字符就可以了。

上代码：

```
#include<bits/stdc++.h>
using namespace std;
int main()
{
  string s;//定义一个字符串
  cin>>s;//输入
  int i;
  for(i=0;i<s.size();i++)//一位一位的判断
  {
  	if(s[i]=='O'||s[i]=='D')//如果这个字符等于O或D
  	  s[i]='0';//就把这个字符替换成0
  	if(s[i]=='I')//如果这个字符是I
	  s[i]='1';//就把这个字符替换成1
	if(s[i]=='Z')//如果这个字符是Z
	  s[i]='2';//就把这个字符替换成2
	 if(s[i]=='S')//如果这个字符是S
	   s[i]='5';//就把这个字符替换成5
	 if(s[i]=='B')//如果这个字符是B
	   s[i]='8';//就把这个字符替换成8    
  }
  cout<<s<<endl;//最后输出，要换行
  return 0;
} 
```
