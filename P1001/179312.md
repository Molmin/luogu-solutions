```
#include<cstdio>
using namespace std;//前三行对于新手题目来说都是标准程序。
int main()
{
  int a,b,s;//定义三个整型的数
  scanf("%d%d",&a,&b);输入要相加的a与b两值
  s=a+b;//计算
  printf("%d",s);//输出结果s
  return 0;//最后一定要return 0哦
 }
```
