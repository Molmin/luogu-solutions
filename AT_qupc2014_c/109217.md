又是一道AT的~~大水~~题。

这题我们只要把不是 '*' 的字符的坐标都存起来，再**循环**、**判断**和**输出**即可。

详细见代码，原版防抄，内含注释，~~不用在意码风~~：

------------

```cpp
#<iostream>//输入输出头文件
using namespace std；//名字空间
long long b[3010][2],d,e,f,g,h,i,j,k,l,m,n,o,p,q,r,s,t,u,v,w,x,y,z;//不用在意这个
char a[1010][1010],c[3010]；
main()
{
	cin>>n>>m>>q；//输入数据
	for(i=0;i<n;i++）//输入数据
	for(k=0;k<m;k++）
	{
		cin>>a[i][k];
		if(a[i][k]!='*'）//如果不是'*'号，就一定是字母，把坐标记下来
		{
			b[a[i][k]][0]=i+1;//在这里字符型会自动转成整型
			b[a[i][k]][1]=k+1;
		}
	}
	for(i=0;i<q;i++)//输入数据
	cin>>c[i]；
	for(i=0;i<q;i++)
	{
		if(b[c[i]][0]!=0)//如果不是零，即有该地点，则输出
		{
			cout<<b[c[i]][0]<<' '<<b[c[i]][1]<<endl;
		}
		if(b[c[i]][0]==0)//没有则输出"NA"
		{
			cout<<"NA"<<endl;
		}
	}
	return 0；
}
```
