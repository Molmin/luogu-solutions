这一题并不难，就是简单的模拟。

```cpp
#include<cstdio>
using namespace std;
int main()
{ 
    int m=0,t,n,s=0,i;int a;
    bool enough=true;//用来判断有没有出现负钱数
    for(i=1;i<=12;i++)//循环读入
        { 
            scanf("%d",&t);
            m=m+300-t;//计算这个月加上300元钱后再用去预算的钱数的钱数。
            if(m>=0)//必须是m大于0才能运算剩余钱数有没有100元或超过100元
        {
            n=m/100;
            m=m-n*100;
            s=s+100*n;
        }
        else//如果出现负钱数就直接输出
        {
            printf("-%d",i);
            enough=false;
            break;
        }
        }
        if(enough==true)//如果没有出现负钱数，就用现在的钱数+存在妈妈那里的钱数+利息，输出
            {
                a=s*1.2+m;
                printf("%d",a);
            }
    return 0;
}
```
