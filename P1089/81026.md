1. 用数组去做，会更容易。
1. 可以选择先输入，再运算，可以解决一些不用输入的时候就退出的烦恼了。
1. 在2的条件下便可以直接跳出循环。
1. 因为存入妈妈手中的钱数是整百的，所以不需要考虑浮点数。
1. cin可以用scanf代替会更高效。
------------

```cpp
#include <bits/stdc++.h>//万能头文件，无须解释。
using namespace std;
int main()
{
	int a[13], m = 300, e = 0, b = 0;//a数组是每月的预算，m为每月给的钱，e为手中的钱，b为预存的钱。
	for (int i = 1; i < 13; i++) {
		cin >> a[i];//输入每个月的预计，方便在实际运算过程中可以直接跳出循环，节省一小部分的时间。
	}
	for (int i = 1; i < 13; i++) {
		e = e + m;//每月给的钱累加。
		e = e - a[i];//减去预算。
		if (e < 0) {//直接判断有没有没钱了。
			cout << -i;//输出月份。
			return 0;//直接结束不用再进行运算。
		}
		else {
			if (e / 100 > 0) {//判断能不能存钱。
				b = b + e / 100 * 100;//能便存钱
				e = e % 100;//更新手中的钱数。
			}
		}
	}
	e = e + b * 120 / 100;//加上预存的钱数。
	cout << e;//输出钱数。
	return 0;//结束。
}
```
