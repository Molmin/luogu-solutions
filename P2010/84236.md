#### 这题是一道简（恶）单（心）的模拟题
1. 输入字符串并转为数字
1. 枚举日期（注意起始日期和闰年）
1. 再转为字符串判断是否回文

AC代码如下

注意一下字符串转数字是怎么转的

## 123456
## 1=123456/100000%10
## 2=123456/10000%10...以此类推

还有时间的计算！

从开始年到结束年枚举年，枚举月时，如果是第一年的话，是从起始月开始枚举的，如果不是，就从1月开始，枚举日时如上，要注意第一年第一月的起始日。

leap函数是很常用的，有需要的可以复制。

判断回文：
当str[i]!=str[8-i-1]时，返回0，表示不是会文，返回1则是回文。

AC代码，可供借鉴：

补充： 当i==e.y&&j==e.m&&t==e.d时，枚举结束，直接输出答案即可

```cpp
#include<bits/stdc++.h>
using namespace std;
string a,b;
int ans;
int leap(int y)
{
    if(y%4==0&&y%100!=0||y%400==0)return 1;
    return 0;
}
int check(int yy,int mm,int dd)
{
	char aa[11];
	aa[0]=yy/1000+'0';
	aa[1]=yy/100%10+'0';
	aa[2]=yy/10%10+'0';
	aa[3]=yy%10+'0';
	aa[4]=mm/10+'0';
	aa[5]=mm%10+'0';
	aa[6]=dd/10+'0';
	aa[7]=dd%10+'0';
	for(int i=0;i<8;++i)
	{
		if(aa[i]!=aa[7-i])
		{
			return 0;
		}
	}
	return 1;
}
int month[13]={20070822,31,28,31,30,31,30,31,31,30,31,30,31};
struct riqi
{
	int y;
	int m;
	int d;
}s,e;
int main()
{
	cin>>a>>b;
	s.y=(a[0]-'0')*1000+(a[1]-'0')*100+(a[2]-'0')*10+(a[3]-'0');
	s.m=(a[4]-'0')*10+(a[5]-'0');
	s.d=(a[6]-'0')*10+(a[7]-'0');
	e.y=(b[0]-'0')*1000+(b[1]-'0')*100+(b[2]-'0')*10+(b[3]-'0');
	e.m=(b[4]-'0')*10+(b[5]-'0');
	e.d=(b[6]-'0')*10+(b[7]-'0');
	for(int i=s.y;i<=e.y;++i)
	{
		if(i%4==0&&i%100!=0||i%400==0)month[2]++;
		for(int j=(i==s.y?s.m:1);j<=12;++j)
		{
			for(int t=(i==s.y&&j==s.m?s.d:1);t<=month[j];++t)
			{
				if(check(i,j,t))ans++;
				if(i==e.y&&j==e.m&&t==e.d)
				{
					cout<<ans;
					return 0;
				}
			}
		}
		if(i%4==0&&i%100!=0||i%400==0)month[2]--;
	}
}
```
