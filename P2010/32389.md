大家好，今天的题解是P2010题，回文日期。其实当我刚刚看到这道题的时候，我是惊呆的，一开始以为要一个日期一个日期的模拟，但转念一想，其实不用，因为每一年中只有1个回文的日期，我们把它弄出来之后只要判断一下它是不是一个合法的日期就可以了。但是这个程序也不是很简单，请大家耐心审看~
```cpp
#include<iostream>
#include<string>
using namespace std;
string s,c;
int a[9],b[9],sum,n,d[13]={0,31,28,31,30,31,30,31,31,30,31,30,31},e[9];//定义一个月份数组，里面储存着每个月的天数，用来比较时使用。
int rn(int n)
{
	if (n%4==0&&n%100!=0||n%400==0) return 1;//这里是一个判断闰年的过程，题干中已经提到，我就不多讲了。
	return 0;
}
int pd (int a1,int a2,int a3)
{
	if (a1>b[1]*1000+b[2]*100+b[3]*10+b[4]) return 0;//这里是一个否定的环节，主要用法为：判断年份是否大于最上限，return 0；判断月份是否大于12，return 0；判断日期是否合法，不合法return 0。
	if (a2>12) return 0;
	if (a3>d[a2%13]) return 0;//这里的a2%13主要作用为派出数组越界的现象。
	return 1;	
}
int main()
{
	cin>>s>>c;
	for (int i=1;i<=8;++i)
	{
		a[i]=s[i-1]-'0';//处理字符串ing······
		b[i]=c[i-1]-'0';
	}
	for (int i=a[1]*1000+a[2]*100+a[3]*10+a[4];i<=b[1]*1000+b[2]*100+b[3]*10+b[4]+2;++i)//这里有点麻烦，其实可以理解为一个年份的上下限，为最小限制和最大限制+2（防止没有枚举到）。
	{
		if (rn(i))//判断闰年，那么二月天数++；
		{
			d[2]++;
			n=1;//标记，后面删除++时用。
		}
		e[4]=i%10;e[3]=i/10%10;e[2]=i/100%10;e[1]=i/1000;
		e[5]=e[4];e[6]=e[3];e[7]=e[2];e[8]=e[1];
		if (pd(i,e[5]*10+e[6],e[7]*10+e[8]))    
		sum++;
		if (n) d[2]--;//判断之前是否++过，否则减去。
		n=0;//去掉++标记，省的出事~
	}
	cout<<sum;
	return 0;//程序结束。
}
```
