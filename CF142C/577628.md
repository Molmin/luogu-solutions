数据范围比较小，于是可以直接手打。

模板程序（生成程序框架）：
```cpp
#include<iostream>
#include<cstdio>
using namespace std;
int main(){
	for(int i=3;i<=9;i++){
		for(int j=3;j<=9;j++){
			cout<<"\tif(n=="<<i<<"&&m=="<<j<<"){\n";
			//'\t'相当于tab，'\n'相当于换行
			for(int x=0;x<=i;x++){
				cout<<"\t\tputs(\"";//'\"'双引号
                if(x!=0)
				    for(int y=1;y<=j;y++) cout<<'.';
				cout<<"\");\n";
			}
			cout<<"\t}\n";
		}
	} 
	return 0;
}
```
