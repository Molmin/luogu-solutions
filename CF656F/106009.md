## 真是一道~~大大大~~的~~水题~~
题意就是：A变成1，1变成10，用计数器加各位数就行了   
上代码~~加注释~~
```cpp
#include<bits/stdc++.h>//万能头文件 
using namespace std;
char a;//定义一个字符 
int b,ans1,c[10],d;//定义一个数，一个计数器，一个数组 
int main()
{
	cin>>a>>b;//输入字符和数 
	if(a=='A')//如果字符等于A
	{
		ans1=1;//计数器定义为初值为1
		for(;b>0;b/=10)c[++d]=b%10;//数字拆分 
		for(int i=1;i<=d;i++)//循环 
		{
			if(c[i]==1)  ans1+=10;//如果b里面有1，计数器加10 
			else    ans1+=c[i];//否则就加各个位的数 
		}
	}
	if(a!='A')//如果字符不等于A 
	{
		for(;b>0;b/=10)c[++d]=b%10;//数字拆分 
		for(int i=1;i<=d;i++)//循环 
		{
			if(c[i]==1)  ans1+=10;//如果b里面有1，计数器加10 
			else    ans1+=c[i];//否则就加各个位的数 
		}
	}
	cout<<ans1;//最后输出计数器 

	return 0;//结束
}
```
