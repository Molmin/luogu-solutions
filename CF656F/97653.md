一题大水题

题意很容易明白:

输入一串字符串 将它分割成一个一个字符 然后累加

**PS:如果输入的是A 累加器要加1 输入1则加10**

最后输出各个字符之和

附上代码:

```pascal
var
  s,i:longint;
  a:array[0..1000] of char;
begin
  i:=1;
  read(a[1]);//输入第一个字符(人家输入字符串 我偏偏输入字符)
  while not eoln do//not eoln一般的用法是:一般是你要读一行数据，又不知道数据的个数，只能根据一行是否结束来判断输入的完成,假设要读一些数，但是事先不知道数的个数,可以用not eoln
  begin
    inc(i);//累加个数
    read(a[i]);//读入
  end;
  for i:=1 to i do//从第一个字符开始打表
  begin
    case a[i] of//打表 因为0加起来的数就是0 所以可以不用写
      'A': s:=s+1;//题意:输入A加1
      '1': s:=s+10;//题意:输入1加10
      '2': s:=s+2;//其它数按本身大小加
      '3': s:=s+3;
      '4': s:=s+4;
      '5': s:=s+5;
      '6': s:=s+6;
      '7': s:=s+7;
      '8': s:=s+8;
      '9': s:=s+9;
    end;
  end;
  writeln(s);//输出各个字符之和
end.
                
```
