作为一位磊落汉子，他让我枚举0~41，我偏不枚举，我就是要用循环一个一个比，我就是【哔】的要把它们一个一个%42然后去重！
```cpp
#include<iostream>
using namespace std;
int main(){
    int a[10],b[10],count=10;//我就不用布尔型变量，初始答案值为10，如有不同就减
    for(int i=0;i<10;i++){
        cin>>a[i];
        a[i]%=42;
        b[i]=1;//做记号
    }
    for(int i=0;i<10;i++){
        for(int j=0;j<10;j++){
            if(a[i]==a[j]&&b[j]&&j!=i){count--;b[j]=0;b[i]=0;}//比过就不比了
        }
    }
    cout<<count;
    return 0;
}
```
