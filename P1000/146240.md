# 题解
本题需要注意：
1. 空格不能少（一个少了会影响全部）
1. 每一行末尾都要记得换行（不然输出会连在一起）

剩下的就看代码了：
```
#include<bits/stdc++.h>//万能头文件
using namespace std;
int main(){
 	cout<<"                ********\n";//空格别少（再次强调）
 	cout<<"               ************\n";//别忘记换行（强调）
	cout<<"               ####....#.\n";
 	cout<<"             #..###.....##....\n";
 	cout<<"             ###.......######              ###            ###\n";
 	cout<<"                ...........               #...#          #...#\n";
 	cout<<"               ##*#######                 #.#.#          #.#.#\n";
 	cout<<"            ####*******######             #.#.#          #.#.#\n";
 	cout<<"           ...#***.****.*###....          #...#          #...#\n";
 	cout<<"           ....**********##.....           ###            ###\n";
 	cout<<"           ....****    *****....\n";
 	cout<<"             ####        ####\n";
 	cout<<"           ######        ######\n";
 	cout<<"##############################################################\n";
 	cout<<"#...#......#.##...#......#.##...#......#.##------------------#\n";
 	cout<<"###########################################------------------#\n";
 	cout<<"#..#....#....##..#....#....##..#....#....#####################\n";
 	cout<<"##########################################    #----------#\n";
 	cout<<"#.....#......##.....#......##.....#......#    #----------#\n";
 	cout<<"##########################################    #----------#\n";
 	cout<<"#.#..#....#..##.#..#....#..##.#..#....#..#    #----------#\n";
 	cout<<"##########################################    ############\n";
 return 0; //完美收尾
} 
```
题解求通过QWQ
