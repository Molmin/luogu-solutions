#include <iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int result[6]={0};
    for(int i=0;i<n;i++)
    {
        double high,low;
        cin>>high>>low;
        if(high>=35.0)
            result[0]++;
        if(high>=30.0 && high<35.0)
            result[1]++;
        if(high>=25.0 && high<30.0)
            result[2]++;
        if(low>=25.0)
            result[3]++;
        if(low<0.0 && high >=0.0)
            result[4]++;
        if(high<0.0)
            result[5]++;
    }
    cout<<result[0]<<" "<<result[1]<<" "<<result[2]<<" "<<result[3]<<" "<<result[4]<<" "<<result[5]<<endl;
    return 0;
}
