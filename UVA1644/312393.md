#include <bits/stdc++.h>
using namespace std;

int cnt=1;

bool pr(int x)
{
	if(x==2||x==3)return 1;
	else if(x%6!=1&&x%6!=5)return 0;
	else
	{
		int n=sqrt(x);
		for(int i=5; i<=n; i+=6)if(x%i==0||x%(i+2)==0)return 0;
		return 1;
	}
}

int main()
{
	printf("2,");
	for(int i=3; i*i<=1299709; i+=2)if(pr(i))printf("%d,",i),cnt++;
	printf("\n%d",cnt);
	return 0;
}
