###### 蒟蒻提示：改变一个字母只会使最终答案加一、不变或减一
```cpp
#include<iostream>
using namespace std;
int main()
{
	int y,m=0;
	bool w=0;
	string n;
	cin>>y>>n;
	for(int i=0;i<y-1;i++)
	{
		if(n[i]=='V'&&n[i+1]=='K')
		m++;
	}
	for(int i=0;i<y-1;i++)
	{
		if(i<y-2)
		if((n[i]=='V')&&(n[i+1]=='V')&&(n[i+2]=='V')||(n[i]=='K')&&(n[i+1]=='K')&&(n[i+2]=='K'))
		w=1;
		if((n[i]=='V')&&(n[i+1]=='V')&&(i==y-2))
		w=1;
		if((n[i]=='K')&&(n[i+1]=='K')&&(i==0))
		w=1;//判定可以改变一个字母使答案加一的三种情况。
	}
	if(w==1)
	m++;
	cout<<m;
	return 0;
}
```
