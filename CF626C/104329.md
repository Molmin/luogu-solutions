```cpp
#include<iostream>
#include<cstdio>
using namespace std;
int lf=1,rg=5000000,n,m,md=(lf+rg)/2,ans;
int main()
{
	scanf("%d%d",&n,&m);
	while(rg-lf>1)
	{
		int k=0,s=1;
		int p=2*n,q=3*m;
		if(p>md)
		{
			s=0;
		}
		int pp=0,qq=0;
		if(p<md&&s)
		{
			pp=(md-p)/2;
		}
		if(q>md&&s)
		{
			s=0;
		}
		if(q<md&&s)
		{
			qq=(md-q)/3;
		}
		k=pp+qq;
		if(s)
		{
			if(k>=md/6)
			{
				s=1;
			}
			else
			{
				s=0;
			}
		}
		if(s==1)
		{
			rg=md;
			md=(lf+rg)/2;
		}
		else
		{
			lf=md;
			md=(lf+rg)/2;
		}
	}
	int k=0,s=1;
	int p=2*n,q=3*m;
	if(p>lf)
	{
		s=0;
	}
	int pp=0,qq=0;
	if(p<lf&&s)
	{
		pp=(lf-p)/2;
	}
	if(q>lf&&s)
	{
		s=0;
	}
	if(q<lf&&s)
	{
		qq=(lf-q)/3;
	}
	k=pp+qq;
	if(s)
	{
		if(k>=lf/6)
		{
			s=1;
		}
		else
		{
			s=0;
		}
	}
	if(s==1)
	{
		ans=lf;
	}
	else
	{
		ans=rg;
	}
	printf("%d",ans);
}
```
