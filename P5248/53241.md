```
#include<bits/stdc++.h>

using namespace std;

long long ans[100010],cnt;

int main()
{
    long long a,b;
    cin>>a>>b;
    while(b)
    {
        ans[++cnt]=b%a;
        b/=a;
    }
    cout<<cnt<<'\n';
    for(int i=1;i<=cnt;i++)
    {
        cout<<ans[i]<<' ';
    }
}
```
