using namespace std; //不打这个不识别cin和cout

{
	char n;//要char！不能int

    for(i=0;i<8;i++)//划重点！在这里等号不取到（在这卡了5分钟）
    { 
	  cin>>n;//输入n
      //核心代码（如果在判断时发现一个1就把计数器+1
      if(n=='1')//判断，来找1

	    ans++;//计数器++（有一个1就+1
	  } 
	}
    cout<<ans;//最后输出ans，也就是有几个1
    return 0;//养成一个良好的习惯
}
