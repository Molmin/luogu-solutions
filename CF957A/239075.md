- 本来这道题我想傻傻的写暴力的，结果写一半发现根本不用，现在开始说推理过程
- 首先对于原本就有两个相同挨着的元素的字符串肯定是 $NO$ ，其余至少都有一种排列方式
- 排除上述情况我们把剩下情况分为几类，我们设 $a,b$ 是任意一种颜色且两者不同
	1. $a?a$ 这时 $?$ 有可能有两种取值
	2. $a?b$ 这时 $?$ 有一种取值
  	3. $a??b$ 或 $a??a$ 无论如何都至少有两种情况，所以可见只要有连续问号就有至少两种可能
  	4. 有 $?$ 在字符串首尾，这样也是至少有两种情况
- 综上所述，只需在无连续相同元素的时候出现 $2,3,4$ 情况就能输出YES
- ~~QED~~

```cpp
#include<bits/stdc++.h>//懒人万能库
using namespace std;
int n;//长度
string s;//字符串
bool flag;//判断变量
int main()
{
	scanf("%d",&n);//喜闻乐见的读入
	cin>>s;
	for(int i=1;i<n-1;i++)//循环开始，首尾各空一格
	{
		if(s[i]!='?'&&(s[i]==s[i-1]||s[i]==s[i+1]))//如果出现连续相同元素就输出
		{
			printf("NO");
			return 0;
		}
		if(s[i]=='?'&&(s[i+1]=='?'||s[i-1]=='?'||s[i-1]==s[i+1]))//判断2,3，情况
			flag=1;
	}
	if(flag||(!flag&&(s[0]=='?'||s[n-1]=='?')))//特判？在字符串首尾的情况
		printf("YES");
	else
		printf("NO");
    return 0;//程序拜拜
}
```
