#include<bits/stdc++.h>

	for(int i=x;i<=y;i++)//10以内
	{
		if(i<10)

		}
		if(i>=100&&i<1000)//100到1000

			if(i/100==2)sum++;//百
		}
		if(i>=1000&&i<10000)//1000到10000
		{
			if(i%10==2)sum++;//个
