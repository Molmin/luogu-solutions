```cpp
#include <cstdio>//头文件
int main()//主函数
{
    int l=0,r=0,t=0,sum=0,p=0;//定义变量
    scanf("%d %d",&l,&r);//读入左边界和右边界（即l和r）
    for(int i=l;i<=r;i++)//从左边界枚举到右边界
    {
        p=i;//p为当前枚举的数
        while(p!=0)//如果p不为0，那么就进循环
        {
            t=p%10;//取p的最后一位数
            p/=10;//舍弃掉当前的最后一位数
            if(t==2)//如果当前的p的最后一位数是2
            {
                sum++;//将2出现的次数累加1
            }
        }
    }
    printf("%d",sum);//输出在l到r之间2出现的次数
    return 0;
}
```
