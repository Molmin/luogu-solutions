### 最后一定注意模拟要从0开始，改变你数组喜欢从1开始的习惯（至少我是），因为取模可能为0；

```cpp
#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
struct aaa{
	int h;//朝向 
	string name;//名字 
}f[100001];  //结构体存 
int main()
{
	int n,m;
	cin>>n>>m; 
	for(int i=0;i<n;i++)
	cin>>f[i].h>>f[i].name;
	int a,b;
	int g=0;//从零开始！从零开始！！从零开始！！！ 
	for(int i=1;i<=m;i++)
	{
		cin>>a>>b;  //读入 
		 if(f[g].h==0&&a==0) //朝里向左 
		 g=(g+n-b)%n;
        else 
		if(f[g].h==0&&a==1)  //朝里向右 
		g=(g+b)%n;
        else  
		if(f[g].h==1&&a==0)//朝外向左 
		g=(g+b)%n;
        else 
		if(f[g].h==1&&a==1)   //朝外向右 
		g=(g+n-b)%n;
	}
	cout<<f[g].name;  //完事！ 
	return 0; 
} 
```
