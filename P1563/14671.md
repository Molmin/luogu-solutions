这种水题还是很容易的模拟直接过。

```cpp
#include<stdio.h>
int main()
{
    int n,m,a[100001],k=1,i,j,x,y;            //定变量  
    char c[100001][11];                      //定数组，用二维数组，以数字字符串来统计 
    scanf("%d %d",&n,&m);                    //输入变量，字符型用%c，整型用%d，浮点型用%f 
    for(i=1;i<=n;i++)                        // 循环的开始
    scanf("%d %s",&a[i],c[i]);               // 给一开始的人数还有朝向定位 
    for(j=1;j<=m;j++)
    {
        scanf("%d %d",&x,&y);                //边进行边输入边处理不用数组减少站位 
        if(a[k]==0)                          //判断朝向内与外，x表示左或者右（具体题目中有） 
        {
            if(x==0)k=k-y;                   //根据要求的寻找进行 ，先不处理范围，最后在判断 
            else k=k+y;
        }
        else
        {
            if(x==0)k=k+y;
            else k=k-y;
        }
        if(k>n)k-=n;                        //上文介绍的判断范围 
        if(k<=0)k+=n;
    }
    printf("%s",c[k]);                      //输出谁拿了爸爸的眼睛 
}
```
